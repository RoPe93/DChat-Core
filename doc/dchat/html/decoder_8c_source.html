49 int
50 decode_header(dchat_pdu_t* pdu, char* line)
51 {
52  char* key; // header key (e.g. Content-Type)
53  char* value; // header value (e.g. text/plain)
54  char* delim = ":"; // delimiter char
55  char* save_ptr; // used for strtok
56  int len; // length of value
57  int end; // index of termination chars (\r)\n of value
58  int ret; // return of parsed value
59  dchat_v1_t proto; // DChat V1 headers
60 
61  // split line: header format -> key:value
62  if ((key = strtok_r(line, delim, &save_ptr)) == NULL)
63  {
64  return -1;
65  }
66  else if ((value = strtok_r(NULL, delim, &save_ptr)) == NULL)
67  {
68  return -1;
69  }
70 
71  // first character must be a whitespace
72  if (strncmp(value, " ", 1) != 0)
73  {
74  return -1;
75  }
76 
77  // skip " "
78  value++;
79 
80  if ((ret = is_valid_termination(value)) == -1)
81  {
82  return -1;
83  }
84 
85  // remove termination characters
86  value[ret] = '\0';
87 
88  if (init_dchat_v1(&proto) == -1)
89  {
90  return -1;
91  }
92 
93  // iterate through headers and check if value is valid
94  // if valid parse value and set attributes in the PDU
95  for (int i = 0; i < HDR_AMOUNT; i++)
96  {
97  if (!strcmp(key, proto.header[i].header_name))
98  {
99  return proto.header[i].str_to_pdu(value, pdu);
100  }
101  }
102 
103  return -1;
104 }

115 int
116 read_line(int fd, char** line)
117 {
118  char* ptr; // line pointer
119  char* alc_ptr = NULL; // used for realloc
120  int len = 1; // current length of string (at least 1 char)
121  int ret; // return value
122  *line = NULL;
123 
124  do
125  {
126  // allocate memory for new character
127  alc_ptr = realloc(*line, len + 1);
128 
129  if (alc_ptr == NULL)
130  {
131  if (*line != NULL)
132  {
133  free(*line);
134  }
135 
136  fatal("Reallocation of input string failed!");
137  }
138 
139  *line = alc_ptr;
140  ptr = *line + len - 1;
141  len++;
142  }
143  while ((ret = read(fd, ptr, 1)) > 0 && *ptr != '\n');
144 
145  // on error or EOF of read
146  if (ret <= 0)
147  {
148  free(*line);
149  return ret;
150  }
151 
152  // terminate string
153  *(ptr + 1) = '\0';
154  return len - 1; // length of string excluding \0
155 }

167 int
168 read_pdu(int fd, dchat_pdu_t* pdu)
169 {
170  char* line; // line read from file descriptor
171  char* contentp; // content pointer
172  int ret; // return value
173  int b; // amount of bytes read as content
174  int len = 0; // amount of bytes read in total
175  // zero out structure
176  memset(pdu, 0, sizeof(*pdu));
177 
178  // read each line of the received pdu
179  if ((ret = read_line(fd, &line)) == -1 || !ret)
180  {
181  return ret;
182  }
183 
184  if (decode_header(pdu, line) == -1 || pdu->version != DCHAT_V1)
185  {
186  free(line);
187  return -1;
188  }
189 
190  len += strlen(line);
191  free(line);
192 
193  // read header lines from file descriptors, until
194  // an empty line is received
195  while ((ret = read_line(fd, &line)) != -1 || ret == 0)
196  {
197  len += strlen(line);
198 
199  // decode read line as header
200  if (decode_header(pdu, line) == -1)
201  {
202  // if line is not a header, it must be an empty line
203  if (!strcmp(line, "\n") || !strcmp(line, "\r\n"))
204  {
205  break; // All headers have been read
206  }
207 
208  ret = -1;
209  break;
210  }
211 
212  free(line);
213  }
214 
215  if (ret <= 0)
216  {
217  free(line);
218  return ret;
219  }
220 
221  // has content type, onion-id and listen-port been specified?
222  if (pdu->content_type == 0 || pdu->onion_id == NULL || pdu->lport == 0)
223  {
224  free(line);
225  return -1;
226  }
227 
228  // allocate memory for content
229  pdu->content = malloc(pdu->content_length + 1);
230  contentp = pdu->content; // point to the beginning
231 
232  // read content frm file descriptor
233  // read x bytes defined by Content-Length
234  for (b = 0, contentp = pdu->content; b < pdu->content_length;
235  b++, contentp++, len++)
236  {
237  if ((ret = read(fd, contentp, 1)) == -1 || !ret)
238  {
239  free(pdu->content);
240  return ret;
241  }
242  }
243 
244  *contentp = '\0'; // NULL terminate potential string
245  return len; // amount of bytes read as content == content length
246 }

259 char*
260 encode_header(dchat_pdu_t* pdu, int header_id)
261 {
262  dchat_v1_t proto; // DChat V1 headers
263  char* header = NULL; // header key string
264  char* value = NULL; // header value string
265  char* line = NULL; // concetenated header, value pair
266  int len; // length of string in total
267  int ret;
268 
269  if (init_dchat_v1(&proto) == -1)
270  {
271  return NULL;
272  }
273 
274  // iterate through supported headers
275  for (int i = 0; i < HDR_AMOUNT; i++)
276  {
277  if (proto.header[i].header_id == header_id)
278  {
279  header = proto.header[i].header_name;
280  len = strlen(header);
281  ret = proto.header[i].pdu_to_str(pdu, &value);
282 
283  if (ret == -1)
284  {
285  return NULL;
286  }
287 
288  len += strlen(value);
289  break;
290  }
291  }
292 
293  len += 4; // add three bytes for ':', a " ", '\n' and '\0';
294 
295  // allocate memory for header string
296  if ((line = malloc(len)) == NULL)
297  {
298  fatal("Memory allocation for header-value string failed!");
299  }
300 
301  // assemble header string
302  line[0] = '\0';
303  strncat(line, header, strlen(header));
304  strncat(line, ":", 1); // seperate key from value -> "key:value"
305  strncat(line, " ", 1); // add a " " after the semicolon -> "key: value"
306  strncat(line, value, strlen(value)); // add value
307  strncat(line, "\n", 1);
308 
309  if (ret == 1) // if value has been allocated dynamically
310  {
311  free(value);
312  }
313 
314  // return header string
315  return line;
316 }

330 int
331 write_pdu(int fd, dchat_pdu_t* pdu)
332 {
333  char* version; //Version of DCHAT
334  char* content_type; //Content-Type header string
335  char* content_length; //Conent-Length header string
336  char* onion_id; //Onion address header string
337  char* lport; //Listening-port header string
338  char* nickname = NULL; //Nickname string
339  char* pdu_raw; //Final PDU
340  int ret; //Return value
341  int pdulen=0; //Total length of PDU
342 
343  //get version string
344  if ((version = encode_header(pdu, HDR_ID_VER)) == NULL)
345  {
346  return -1;
347  }
349  else if ((content_type = encode_header(pdu, HDR_ID_CTT)) == NULL)
350  {
351  free(version);
352  return -1;
353  }
354  //get Content-Length string
355  else if ((content_length = encode_header(pdu, HDR_ID_CTL)) == NULL)
356  {
357  free(version);
358  free(content_type);
359  return -1;
360  }
361  //get Listen-Port string
362  else if ((lport = encode_header(pdu, HDR_ID_LNP)) == NULL)
363  {
364  free(version);
365  free(content_type);
366  free(content_length);
367  return -1;
368  }
369  //get Onion-ID string
370  else if ((onion_id = encode_header(pdu, HDR_ID_ONI)) == NULL)
371  {
372  free(version);
373  free(content_type);
374  free(content_length);
375  free(lport);
376  return -1;
377  }
378 
379  //get Nickname string
380  if ((nickname = encode_header(pdu, HDR_ID_NIC)) == NULL)
381  {
382  free(version);
383  free(content_type);
384  free(content_length);
385  free(lport);
386  free(onion_id);
387  return -1;
388  }
389 
390  pdulen = strlen(version);
391  pdulen += strlen(content_type);
392  pdulen += strlen(content_length);
393  pdulen += strlen(onion_id);
394  pdulen += strlen(lport);
395  pdulen += strlen(nickname);
396  pdulen += 1; //for empty line
397  pdulen += pdu->content_length;
398  //allocate memory for pdu
399  pdu_raw = malloc(pdulen + 1);
400 
401  if (pdu_raw == NULL)
402  {
403  fatal("Memory allocation for PDU failed!");
404  }
405 
406  //craft pdu
407  pdu_raw[0] = '\0';
408  strncat(pdu_raw, version, strlen(version));
409  strncat(pdu_raw, content_type, strlen(content_type));
410  strncat(pdu_raw, content_length, strlen(content_length));
411  strncat(pdu_raw, onion_id, strlen(onion_id));
412  strncat(pdu_raw, lport, strlen(lport));
413  strncat(pdu_raw, nickname, strlen(nickname));
414  strncat(pdu_raw, "\n", 1);
415  strncat(pdu_raw, pdu->content, pdu->content_length);
416  //write pdu to file descriptor
417  ret = write(fd, pdu_raw, pdulen);
418  free(content_type);
419  free(content_length);
420  free(onion_id);
421  free(lport);
422  free(pdu_raw);
423  return ret;
424 }
