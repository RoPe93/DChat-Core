50 int
51 decode_header(dchat_pdu_t* pdu, char* line)
52 {
53  char* temp;
54  char* key; // header key (e.g. Content-Type)
55  char* value; // header value (e.g. text/plain)
56  char* delim = ":"; // delimiter char
57  char* save_ptr; // used for strtok
58  int len; // length of value
59  int end; // index of termination chars (\r)\n of value
60  int ret; // return of parsed value
61  dchat_v1_t proto; // DChat V1 headers
62 
63 
64  if(line == NULL){
65  return -1;
66  }
67 
68  // copy line to work with
69  temp = malloc(strlen(line) + 1);
70  if(temp == NULL){
71  fatal("Memory allocation for temporary decoder line failed!");
72  }
73  temp[0] = '\0';
74  strcat(temp, line);
75 
76  // split line: header format -> key:value
77  if ((key = strtok_r(temp, delim, &save_ptr)) == NULL)
78  {
79  free(temp);
80  return -1;
81  }
82  else if ((value = strtok_r(NULL, delim, &save_ptr)) == NULL)
83  {
84  free(temp);
85  return -1;
86  }
87  // only split the very first token from temp, to keep
88  // the value one token (containing possible delim chars)
89  if(save_ptr != NULL)
90  {
91  value[strlen(value)] = *delim;
92  }
93 
94  // first character must be a whitespace
95  if (strncmp(value, " ", 1) != 0)
96  {
97  free(temp);
98  return -1;
99  }
100 
101  // skip " "
102  value++;
103 
104  if ((ret = is_valid_termination(value)) == -1)
105  {
106  free(temp);
107  return -1;
108  }
109 
110  // remove termination characters
111  value[ret] = '\0';
112 
113  if (init_dchat_v1(&proto) == -1)
114  {
115  free(temp);
116  return -1;
117  }
118 
119  // iterate through headers and check if value is valid
120  // if valid parse value and set attributes in the PDU
121  for (int i = 0; i < HDR_AMOUNT; i++)
122  {
123  if (!strcmp(key, proto.header[i].header_name))
124  {
125  free(temp);
126  return proto.header[i].str_to_pdu(value, pdu);
127  }
128  }
129 
130  free(temp);
131  return -1;
132 }

143 int
144 read_line(int fd, char** line)
145 {
146  char* ptr; // line pointer
147  char* alc_ptr = NULL; // used for realloc
148  int len = 1; // current length of string (at least 1 char)
149  int ret; // return value
150  *line = NULL;
151 
152  do
153  {
154  // allocate memory for new character
155  alc_ptr = realloc(*line, len + 1);
156 
157  if (alc_ptr == NULL)
158  {
159  if (*line != NULL)
160  {
161  free(*line);
162  }
163 
164  fatal("Reallocation of input string failed!");
165  }
166 
167  *line = alc_ptr;
168  ptr = *line + len - 1;
169  len++;
170  }
171  while ((ret = read(fd, ptr, 1)) > 0 && *ptr != '\n');
172 
173  // on error or EOF of read
174  if (ret <= 0)
175  {
176  free(*line);
177  return ret;
178  }
179 
180  // terminate string
181  *(ptr + 1) = '\0';
182  return len - 1; // length of string excluding \0
183 }

195 int
196 read_pdu(int fd, dchat_pdu_t* pdu)
197 {
198  char* line; // line read from file descriptor
199  char* contentp; // content pointer
200  int ret; // return value
201  int b; // amount of bytes read as content
202  int len = 0; // amount of bytes read in total
203  // zero out structure
204  memset(pdu, 0, sizeof(*pdu));
205 
206  // read each line of the received pdu
207  if ((ret = read_line(fd, &line)) == -1 || !ret)
208  {
209  return ret;
210  }
211 
212  // first header must be version header
213  if (decode_header(pdu, line) == -1 || pdu->version != DCHAT_V1)
214  {
215  ret = -1;
216  }
217 
218  if(ret != -1)
219  {
220  len += strlen(line);
221  free(line);
222 
223  // read header lines from file descriptors, until
224  // an empty line is received
225  while ((ret = read_line(fd, &line)) != -1 || ret == 0)
226  {
227  len += strlen(line);
228 
229  if (decode_header(pdu, line) == -1)
230  {
231  // if line is not a header, it must be an empty line
232  if (!strcmp(line, "\n") || !strcmp(line, "\r\n"))
233  {
234  break; // All headers have been read
235  }
236  ret = -1;
237  break;
238  }
239 
240  free(line);
241  }
242  }
243 
244  // On error print illegal line
245  if(ret == -1)
246  {
247  log_msg(LOG_ERR, "Illegal PDU header received: '%s'", line);
248  }
249  // EOF or ERROR
250  if (ret <= 0)
251  {
252  free(line);
253  return ret;
254  }
255 
256  // has content type, onion-id and listen-port been specified?
257  if (pdu->content_type == 0 || pdu->onion_id == NULL || pdu->lport == 0)
258  {
259  log_msg(LOG_ERR, "Mandatory PDU headers are missing!");
260  free(line);
261  return -1;
262  }
263 
264  // allocate memory for content
265  pdu->content = malloc(pdu->content_length + 1);
266  contentp = pdu->content; // point to the beginning
267 
268  // read content frm file descriptor
269  // read x bytes defined by Content-Length
270  for (b = 0, contentp = pdu->content; b < pdu->content_length;
271  b++, contentp++, len++)
272  {
273  if ((ret = read(fd, contentp, 1)) == -1 || !ret)
274  {
275  free(pdu->content);
276  return ret;
277  }
278  }
279 
280  *contentp = '\0'; // NULL terminate potential string
281  return len; // amount of bytes read as content == content length
282 }

295 int
296 encode_header(dchat_pdu_t* pdu, int header_id, char** headerline)
297 {
298  dchat_v1_t proto; // DChat V1 headers
299  char* header = NULL; // header key string
300  char* value = NULL; // header value string
301  int len; // length of string in total
302  int ret;
303 
304  if (init_dchat_v1(&proto) == -1)
305  {
306  return -1;
307  }
308 
309  // iterate through supported headers
310  for (int i = 0; i < HDR_AMOUNT; i++)
311  {
312  if (proto.header[i].header_id == header_id)
313  {
314  header = proto.header[i].header_name;
315  len = strlen(header);
316  if((ret = proto.header[i].pdu_to_str(pdu, &value)) == -1)
317  {
318  return -1;
319  }
320 
321  // check if header is mandatory, if no value has been set
322  // in the pdu structure
323  if(ret == 1)
324  {
325  // if header is mandatory -> raise error
326  // otherwise just return and do nothing
327  if(proto.header[i].mandatory)
328  {
329  return -1;
330  }
331 
332  return 1;
333  }
334 
335  len += strlen(value);
336  len += 4; // add three bytes for ':', a " ", '\n' and '\0';
337 
338  // allocate memory for header string
339  if ((*headerline = malloc(len)) == NULL)
340  {
341  fatal("Memory allocation for header-value string failed!");
342  }
343 
344  // assemble header string
345  *headerline[0] = '\0';
346  strncat(*headerline, header, strlen(header));
347  strncat(*headerline, ":", 1); // seperate key from value -> "key:value"
348  strncat(*headerline, " ", 1); // add a " " after the semicolon -> "key: value"
349  strncat(*headerline, value, strlen(value)); // add value
350  strncat(*headerline, "\n", 1);
351 
352  // free converted pdu structure value
353  free(value);
354  return 0;
355  }
356  }
357 
358  return -1;
359 }

373 int
374 write_pdu(int fd, dchat_pdu_t* pdu)
375 {
376  dchat_v1_t proto; //Available DChat headers
377  char* header; //DChat header
378  char* pdu_raw; //Final PDU
379  int ret; //Return value
380  int pdulen=0; //Total length of PDU
381 
382  if (init_dchat_v1(&proto) == -1)
383  {
384  return -1;
385  }
386 
387  // get version header string
388  if ((ret = encode_header(pdu, HDR_ID_VER, &header)) == -1 || ret == 1)
389  {
390  return -1;
391  }
392 
393  pdulen += strlen(header);
394  if((pdu_raw = malloc(pdulen + 1)) == NULL){
395  fatal("Memory allocation for pdu failed!");
396  }
397 
398  // copy version header to raw pdu
399  pdu_raw[0] = '\0';
400  strcat(pdu_raw, header);
401  free(header);
402 
403  // iterate through supported headers
404  for (int i = 0; i < HDR_AMOUNT; i++)
405  {
406  // get header strings except version header, if set in pdu structure
407  if(proto.header[i].header_id != HDR_ID_VER){
408  // get header string
409  if ((ret = encode_header(pdu, proto.header[i].header_id, &header)) == -1)
410  {
411  free(pdu_raw);
412  return -1;
413  }
414 
415  // if header was mandatory, but value was not set in pdu structure
416  // raise an error
417  if(ret == 1)
418  {
419  if(proto.header[i].mandatory)
420  {
421  free(pdu_raw);
422  return -1;
423  }
424 
425  continue;
426  }
427 
428  // (re)allocate memory for new header (excluding \0)
429  pdulen += strlen(header);
430  pdu_raw = realloc(pdu_raw, pdulen);
431  if (pdu_raw == NULL)
432  {
433  fatal("Reallocation of pdu failed!");
434  }
435 
436  // copy header to pdu
437  strcat(pdu_raw, header);
438  free(header);
439 
440  }
441  }
442 
443  // (re)allocate memory for empty line and content (excluding \0)
444  pdulen += pdu->content_length + 1;
445  pdu_raw = realloc(pdu_raw, pdulen);
446  if (pdu_raw == NULL)
447  {
448  fatal("Reallocation of pdu failed!");
449  }
450 
451  // add empty line
452  strcat(pdu_raw, "\n");
453  // add content
454  strncat(pdu_raw, pdu->content, pdu->content_length);
455  //write pdu to file descriptor
456  ret = write(fd, pdu_raw, pdulen);
457  free(pdu_raw);
458 
459  return pdulen;
460 }
