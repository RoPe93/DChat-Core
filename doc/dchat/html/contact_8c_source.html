51 send_contacts(dchat_conf_t* cnf, int n)
52 {
53  dchat_pdu_t pdu; // pdu with contact information
54  char* contact_str; // pointer to a string representation of a contact
55  int i;
56  int ret; // return value
57  int pdu_len = 0; // total content length of pdu-packet that will be sent
58  contact_t* contact; // contact that will be converted to a string
59  // initialize PDU
60  init_dchat_pdu(&pdu, 1.0, CTT_ID_DSC, cnf->me.onion_id, cnf->me.lport,
61  cnf->me.name);
62 
63  // iterate through our contactlist
64  for (i = 0; i < cnf->cl.cl_size; i++)
65  {
66  // except client n to whom we sent our contactlist
67  if (n == i)
68  {
69  continue;
70  }
71 
72  // temporarily point to a contact
73  contact = &(cnf->cl.contact[i]);
74 
75  // if its not an empty contact slot and contact is not temporary (has not sent "control/discover" yet)
76  if (contact->lport != 0)
77  {
78  // convert contact to a string
79  if ((contact_str = contact_to_string(contact)) == NULL)
80  {
81  log_msg(LOG_WARN, "Conversion of contact '%s' to string failed! - Skipped",
82  contact->name);
83  continue;
84  }
85 
86  // (re)allocate memory for pdu for contact string
87  pdu.content = realloc(pdu.content, pdu_len + strlen(contact_str) + 1);
88  pdu.content[pdu_len] = '\0'; // set the first byte to \0.. used for strcat
89 
90  // could not allocate memory for content
91  if (pdu.content == NULL)
92  {
93  fatal("Memory reallocation for contactlist failed!");
94  }
95 
96  // add contact information to content
97  strncat(pdu.content, contact_str, strlen(contact_str));
98  // increase size of pdu content-length
99  pdu_len += strlen(contact_str);
100  free(contact_str);
101  }
102  }
103 
104  // set content-length of this pdu
105  pdu.content_length = pdu_len;
106 
107  // write pdu inkluding all addresses of our contacts
108  if ((ret = write_pdu(cnf->cl.contact[n].fd, &pdu)) == -1)
109  {
110  log_msg(LOG_ERR, "Sending of contactlist failed!");
111  }
112 
113  if (pdu.content_length != 0)
114  {
115  free(pdu.content);
116  }
117 
118  return ret;
119 }

132 int
133 receive_contacts(dchat_conf_t* cnf, dchat_pdu_t* pdu)
134 {
135  contact_t contact;
136  int ret = 0; // return value
137  int new_contacts = 0; // stores how many new contacts have been received
138  int known_contacts = 0; // stores how many known contacts have been received
139  int line_begin = 0; // offset of content of given pdu
140  int line_end = 0; // offset (end) of content of given pdu
141  char* line; // contact string
142 
143  // as long as the line_end index is lower than content-length
144  while (line_end < pdu->content_length)
145  {
146  line_begin = line_end;
147 
148  // extract a line from the content of the pdu
149  if ((line_end = get_content_part(pdu, line_begin, '\n', &line)) == -1)
150  {
151  log_msg(LOG_ERR, "Extraction of contact line from received PDU failed!");
152  ret = -1;
153  break;
154  }
155 
156  // parse line ane make string to contact
157  if (string_to_contact(&contact, line) == -1)
158  {
159  log_msg(LOG_WARN, "Conversion of string to contact failed! - Skipped");
160  ret = -1;
161  continue;
162  }
163 
164  // if parsed contact is unknown
165  if (find_contact(cnf, &contact, 0) == -2)
166  {
167  // increment new contacts counter
168  new_contacts++;
169 
170  // connect to new contact, add him as contact, and send contactlist to him
171  if (handle_local_conn_request(cnf, contact.onion_id, contact.lport) == -1)
172  {
173  log_msg(LOG_WARN, "Connection to new contact failed!");
174  ret = -1;
175  }
176  }
177  else
178  {
179  // we found parsed contact in contactlist -> increment known contacts counter
180  known_contacts++;
181  }
182 
183  // increment end of line index, to point to the beginning of the next line
184  line_end++;
185  }
186 
187  return ret != -1 ? new_contacts : -1;
188 }

205 int
206 check_duplicates(dchat_conf_t* cnf, int n)
207 {
208  int fst_oc; // index of first occurance of contact
209  int sec_oc; // index of duplicate
210  contact_t* temp; // temporay contact variable
211  int connect_contact = 0; // index of contact to whom we connected to
212  int accept_contact = 0; // index of contact from whom we accepted a connection
213  int ret;
214  // check if given contact is in the contactlist
215  fst_oc = find_contact(cnf, &cnf->cl.contact[n], 0);
216 
217  // contact is this client
218  if (fst_oc == -1)
219  {
220  return n;
221  }
222 
223  if (fst_oc == -2)
224  {
225  return -1; // contact not in list
226  }
227 
228  // check if given contact is in the contactlist a second time
229  sec_oc = find_contact(cnf, &cnf->cl.contact[n], fst_oc + 1);
230 
231  if (sec_oc == -2)
232  {
233  return -1; // no duplicate contact
234  }
235 
236  // extract port of sockaddr_storage structure
237  temp = &cnf->cl.contact[fst_oc];
238 
239  // which kind of contact has to be deleted?
240  if (temp->accepted)
241  {
242  accept_contact = fst_oc;
243  connect_contact = sec_oc;
244  }
245  else
246  {
247  connect_contact = fst_oc;
248  accept_contact = sec_oc;
249  }
250 
251  // if local onion address is greater than the remote one
252  // than the index of the contact, who got added because of a "connect",
253  // will be returned
254  ret = strcmp(cnf->me.onion_id, cnf->cl.contact[n].onion_id);
255 
256  if (ret > 0)
257  {
258  return connect_contact;
259  }
260  // otherwise it is the other way round
261  else if (ret < 0)
262  {
263  return accept_contact;
264  }
265  // if ip addresses are equal, do the same for the listening port
266  else if (cnf->me.lport > cnf->cl.contact[n].lport)
267  {
268  return connect_contact;
269  }
270  else if (cnf->me.lport < cnf->cl.contact[n].lport)
271  {
272  return accept_contact;
273  }
274  else
275  {
276  return accept_contact;
277  }
278 }

288 char*
289 contact_to_string(contact_t* contact)
290 {
291  char* contact_str; // pointer to string repr. of contact
292  char port_str[MAX_INT_STR + 1]; // max. characters of an int
293  int contact_len; // length of contact structure
294 
295  if (!is_valid_onion(contact->onion_id))
296  {
297  return NULL;
298  }
299 
300  if (!is_valid_port(contact->lport))
301  {
302  return NULL;
303  }
304 
305  // convert port to a string
306  snprintf(port_str, MAX_INT_STR, "%u", contact->lport);
307  port_str[5] = '\0';
308  // +4 for two " ", \n and \0
309  contact_len = strlen(contact->onion_id) + strlen(port_str) + 4;
310 
311  // allocate memory for the contact string repr.
312  if ((contact_str = malloc(contact_len)) == NULL)
313  {
314  fatal("Memory allocaction for contact string failed!");
315  }
316 
317  // first byte has to be '\0', otherwise strcat is undefined
318  contact_str[0] = '\0';
319  // create contact string like this:
320  // <ip address> <port>\n
321  // (see: DChat Protocol - Contact Exchange
322  strncat(contact_str, contact->onion_id, strlen(contact->onion_id));
323  strncat(contact_str, " ", 1);
324  strncat(contact_str, port_str, strlen(port_str));
325  strncat(contact_str, "\n", 1);
326  return contact_str;
327 }

340 string_to_contact(contact_t* contact, char* string)
341 {
342  char* onion_id; // onion address string (splitted from line)
343  char* port; // port string (splitted from line)
344  int lport; // converted listening port
345  char* ptr; // used for strtol
346  char* save_ptr; // pointer for strtok_r
347 
348  // split onion-id from string
349  if ((onion_id = strtok_r(string, " ", &save_ptr)) == NULL)
350  {
351  log_msg(LOG_ERR, "Missing onion-id in contact string!");
352  return -1;
353  }
354 
355  // split port from string
356  if ((port = strtok_r(NULL, "\n", &save_ptr)) == NULL)
357  {
358  log_msg(LOG_ERR, "Missing listening port in contact string!");
359  return -1;
360  }
361 
362  if (!is_valid_onion(onion_id))
363  {
364  log_msg(LOG_ERR, "Invalid onion-id of contact string!");
365  return -1;
366  }
367 
368  // could line be splittet into ip and port?
369  // convert port string to integer
370  lport = (int) strtol(port, &ptr, 10);
371 
372  if (ptr[0] != '\0' || !is_valid_port(lport))
373  {
374  log_msg(LOG_ERR, "Invalid port of contact string!");
375  return -1;
376  }
377 
378  contact->lport = lport;
379  contact->onion_id[0] = '\0';
380  strncat(contact->onion_id, onion_id, ONION_ADDRLEN);
381  return 0;
382 }

410

412  if ((new_contact_list = malloc(newsize * sizeof(contact_t))) == NULL)
413  {
414  fatal("Reallocation of contactlist failed!");
415  }
416 
417  // zero out new contactlist
418  memset(new_contact_list, 0, newsize * sizeof(contact_t));
419 
420  // copy contacts to new contactlist
421  for (i = 0; i < cnf->cl.cl_size; i++)
422  {

428  }
429 
430  // set new size and point to new contactlist in the global config
431  cnf->cl.cl_size = newsize;
432  cnf->cl.contact = new_contact_list;
433  // free old contactlist
434  free(old_contact_list);
