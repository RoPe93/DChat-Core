52 int
53 send_contacts(dchat_conf_t* cnf, int n)
54 {
55  dchat_pdu_t pdu; // pdu with contact information
56  char* contact_str; // pointer to a string representation of a contact
57  int i;
58  int ret; // return value
59  int pdu_len = 0; // total content length of pdu-packet that will be sent
60  contact_t* contact; // contact that will be converted to a string
61  // zero out new pdu and set initial dchat headers
62  //FIXME: create function for PDU creation
63  memset(&pdu, 0, sizeof(pdu));
64  pdu.content_type = CT_CTRL_DISC;
65  pdu.content_length = 0;
66  pdu.onion_id = cnf->me.onion_id;
67  pdu.nickname = cnf->me.name;
68  pdu.lport = cnf->me.lport;
69 
70  // iterate through our contactlist
71  for (i = 0; i < cnf->cl.cl_size; i++)
72  {
73  // except client n to whom we sent our contactlist
74  if (n == i)
75  {
76  continue;
77  }
78 
79  // temporarily point to a contact
80  contact = &(cnf->cl.contact[i]);
81 
82  // if its not an empty contact slot and contact is not temporary (has not sent "control/discover" yet)
83  if (contact->lport != 0)
84  {
85  // convert contact to a string
86  if ((contact_str = contact_to_string(contact)) == NULL)
87  {
88  log_msg(LOG_WARN, "send_contacts() - Could not send contact");
89  ret = -1;
90  continue;
91  }
92  // (re)allocate memory for pdu for contact string
93  pdu.content = realloc(pdu.content, pdu_len + strlen(contact_str) + 1);
94  pdu.content[pdu_len] = '\0'; // set the first byte to \0.. used for strcat
95 
96  // could not allocate memory for content
97  if (pdu.content == NULL)
98  {
99  log_msg(LOG_ERR, "send_contacts() - Could not allocate memory");
100  ret = -1;
101  }
102  else
103  {
104  // add contact information to content
105  strncat(pdu.content, contact_str, strlen(contact_str));
106  // increase size of pdu content-length
107  pdu_len += strlen(contact_str);
108  }
109 
110  free(contact_str);
111  }
112  }
113 
114  // set content-length of this pdu
115  pdu.content_length = pdu_len;
116 
117  // write pdu inkluding all addresses of our contacts
118  if ((ret = write_pdu(cnf->cl.contact[n].fd, &pdu)) == -1)
119  {
120  log_msg(LOG_ERR, "send_contacts() failed - Could not send pdu");
121  ret = -1;
122  }
123 
124  if (pdu.content_length != 0)
125  {
126  free(pdu.content);
127  }
128 
129  return ret;
130 }

143 int
144 receive_contacts(dchat_conf_t* cnf, dchat_pdu_t* pdu)
145 {
146  contact_t contact;
147  int ret = 0; // return value
148  int new_contacts = 0; // stores how many new contacts have been received
149  int known_contacts = 0; // stores how many known contacts have been received
150  int line_begin = 0; // offset of content of given pdu
151  int line_end = 0; // offset (end) of content of given pdu
152  char* line; // contact string
153 
154  // as long as the line_end index is lower than content-length
155  while (line_end < pdu->content_length)
156  {
157  line_begin = line_end;
158 
159  // extract a line from the content of the pdu
160  if ((line_end = get_content_part(pdu, line_begin, '\n', &line)) == -1)
161  {
162  log_msg(LOG_ERR, "receive_contacts(): Could not extract line from PDU");
163  ret = -1;
164  break;
165  }
166 
167  // parse line ane make string to contact
168  if (string_to_contact(&contact, line) == -1)
169  {
170  log_msg(LOG_WARN, "receive_contacts(): Could not convert string to contact");
171  ret = -1;
172  continue;
173  }
174 
175  // if parsed contact is unknown
176  if (find_contact(cnf, &contact, 0) == -2)
177  {
178  // increment new contacts counter
179  new_contacts++;
180 
181  // connect to new contact, add him as contact, and send contactlist to him
182  if (handle_local_conn_request(cnf, contact.onion_id, contact.lport) == -1)
183  {
184  log_msg(LOG_WARN,
185  "receive_contacts(): Could not execute connection request successfully");
186  }
187  }
188  else
189  {
190  // we found parsed contact in contactlist -> increment known contacts counter
191  known_contacts++;
192  }
193 
194  // increment end of line index, to point to the beginning of the next line
195  line_end++;
196  }
197 
198  return ret != -1 ? new_contacts : -1;
199 }

216 check_duplicates(dchat_conf_t* cnf, int n)
217 {
218  int fst_oc; // index of first occurance of contact
219  int sec_oc; // index of duplicate
220  contact_t* temp; // temporay contact variable
221  int connect_contact = 0; // index of contact to whom we connected to
222  int accept_contact = 0; // index of contact from whom we accepted a connection
223  int ret;
224  // check if given contact is in the contactlist
225  fst_oc = find_contact(cnf, &cnf->cl.contact[n], 0);
226 
227  // contact is this client
228  if(fst_oc == -1){
229  return n;
230  }
231 
232  if (fst_oc == -2)
233  {
234  return -2; // error contact not in list
235  }
236 
237  // check if given contact is in the contactlist a second time
238  sec_oc = find_contact(cnf, &cnf->cl.contact[n], fst_oc + 1);
239 
240  if (sec_oc == -2)
241  {
242  return -1; // no duplicate contact
243  }
244 
245  // extract port of sockaddr_storage structure
246  temp = &cnf->cl.contact[fst_oc];
247 
248  // which kind of contact has to be deleted?
249  if (temp->accepted)
250  {
251  accept_contact = fst_oc;
252  connect_contact = sec_oc;
253  }
254  else
255  {
256  connect_contact = fst_oc;
257  accept_contact = sec_oc;
258  }
259 
260  // if local onion address is greater than the remote one
261  // than the index of the contact, who got added because of a "connect",
262  // will be returned
263  ret = strcmp(cnf->me.onion_id, cnf->cl.contact[n].onion_id);
264  if (ret > 0)
265  {
266  return connect_contact;
267  }
268  // otherwise it is the other way round
269  else if (ret < 0)
270  {
271  return accept_contact;
272  }
273  // if ip addresses are equal, do the same for the listening port
274  else if (cnf->me.lport > cnf->cl.contact[n].lport)
275  {
276  return connect_contact;
277  }
278  else if (cnf->me.lport < cnf->cl.contact[n].lport)
279  {
280  return accept_contact;
281  }
282  else
283  {
284  return accept_contact;
285  }
286 }

296 char*
297 contact_to_string(contact_t* contact)
298 {
299  char* contact_str; // pointer to string repr. of contact
300  char port_str[MAX_INT_STR + 1]; // max. characters of an int
301  int contact_len; // length of contact structure
302 
303  if(contact->onion_id == NULL)
304  {
305  return NULL;
306  }
307  //FIXME: check valid port
308  // convert port to a string
309  snprintf(port_str, MAX_INT_STR, "%u", contact->lport);
310  port_str[5] = '\0';
311  // +4 for two " ", \n and \0
312  contact_len = strlen(contact->onion_id) + strlen(port_str) + 4;
313 
314  // allocate memory for the contact string repr.
315  if ((contact_str = malloc(contact_len)) == NULL)
316  {
317  log_msg(LOG_ERR, "contact_to_string() failed - Could not allocate memory");
318  return NULL;
319  }
320 
321  // first byte has to be '\0', otherwise strcat is undefined
322  contact_str[0] = '\0';
323  // create contact string like this:
324  // <ip address> <port>\n
325  // (see: DChat Protocol - Contact Exchange
326  strncat(contact_str, contact->onion_id, strlen(contact->onion_id));
327  strncat(contact_str, " ", 1);
328  strncat(contact_str, port_str, strlen(port_str));
329  strncat(contact_str, "\n", 1);
330  return contact_str;
331 }

363

365  if (onion_id != NULL && port != NULL)
366  {
367  // convert port string to integer
368  contact->lport = atoi(port);
369  //FIXME: check valid onion-id
370  contact->onion_id[0] = '\0';
371  strncat(contact->onion_id, onion_id, ONION_ADDRLEN);
372  }

374  return 0;

403

405  if ((new_contact_list = malloc(newsize * sizeof(contact_t))) == NULL)

407  log_msg(LOG_ERR, "realloc_contactlist() failed - Could not allocate memory");
408  return -1;

413 
414  // copy contacts to new contactlist

444 {

446 
447  // if contactlist is full - resize it so that we can store more contacts in it
448  if (cnf->cl.used_contacts == cnf->cl.cl_size)

455 
456  // search for an empty place to store the new contact
457  for (i = 0; i < cnf->cl.cl_size; i++)
458  {
