47 char*
48 get_short_options(cli_options_t* options)
49 {
50  char* opt_str = malloc(CLI_OPT_AMOUNT * 2 + 1); // max. possible string
51 
52  if (opt_str == NULL)
53  {
54  fatal("Memory allocation for short options failed!");
55  }
56 
57  opt_str[0] = '\0';
58 
59  for (int i = 0; i < CLI_OPT_AMOUNT; i++)
60  {
61  // append short options
62  strncat(opt_str, &options->option[i].opt, 1);
63 
64  // getopt uses `:` to specify a required
65  // argument for an option
66  if (options->option[i].mandatory_argument)
67  {
68  strncat(opt_str, ":", 1);
69  }
70  }
71 
72  return opt_str;
73 }

112 int
113 init_cli_options(cli_options_t* options)
114 {
115  int temp_size;
116  // possible commandline options
117  cli_option_t temp[] =
118  {
119  OPTION(CLI_OPT_LONI, CLI_LOPT_LONI, CLI_OPT_ARG_LONI, 1, "Set the onion id of the local hidden service.", loni_parse),
120  OPTION(CLI_OPT_NICK, CLI_LOPT_NICK, CLI_OPT_ARG_NICK, 1, "Set the nickname for this chat session.", nick_parse),
121  OPTION(CLI_OPT_LPRT, CLI_LOPT_LPRT, CLI_OPT_ARG_LPRT, 0, "Set the local listening port.", lprt_parse),
122  OPTION(CLI_OPT_RONI, CLI_LOPT_RONI, CLI_OPT_ARG_RONI, 0, "Set the onion id of the remote host to whom a connection should be established.", roni_parse),
123  OPTION(CLI_OPT_RPRT, CLI_LOPT_RPRT, CLI_OPT_ARG_RPRT, 0, "Set the remote port of the remote host who will accept connections on this port.", rprt_parse),
124  OPTION(CLI_OPT_HELP, CLI_LOPT_HELP, CLI_OPT_ARG_HELP, 0, "Display help.", help_parse)
125  };
126  temp_size = sizeof(temp) / sizeof(temp[0]);
127 
128  if (temp_size > CLI_OPT_AMOUNT)
129  {
130  return -1;
131  }
132 
133  memset(options, 0, sizeof(*options));
134  memcpy(options->option, &temp, sizeof(temp));
135  return 0;
136 }

154 int
155 read_conf(dchat_conf_t* cnf, char* filepath, int* required_set)
156 {
157  FILE* f; // config file file stream
158  int fd; // config file file descriptor
159  char* line; // read line of config file
160  char* opt; // cli opt
161  char* arg; // cli opt argument
162  char* delim = " "; // cli opt delimiter
163  int lctr = 1; // line counter
164  int end; // index of termination
165  int found_opt; // flag if read option has been found
166  int error = 0; // boolean if error occured
167  cli_options_t options; // available command line options
168  int ret;
169 
170  // init available options
171  if (init_cli_options(&options) == -1)
172  {
173  log_msg(LOG_ERR, "Initialization of command line options failed!");
174  return -1;
175  }
176 
177  // open config file
178  f = fopen(filepath, "r");
179 
180  if (f == NULL)
181  {
182  log_errno(LOG_ERR, "Could not read file '%s'!", filepath);
183  return -1;
184  }
185 
186  // get file descriptor of file stream
187  fd = fileno(f);
188 
189  // read confif file line by line
190  while (read_line(fd, &line) > 0)
191  {
192  // split line to get option and option argument
193  if ((opt = strtok_r(line, delim, &arg)) == NULL)
194  {
195  error = 1;
196  break;
197  }
198 
199  // skip spaces if there is an argument
200  if (remove_leading_spaces(arg) != NULL)
201  {
202  // get index of termination char (\n or \r\n)
203  if ((end = is_valid_termination(arg)) == -1)
204  {
205  error = 1;
206  break;
207  }
208 
209  // remove termination char
210  arg[end] = '\0';
211  }
212  else
213  {
214  arg = "";
215  }
216 
217  // check if read option is an supported/available option
218  found_opt = 0;
219 
220  for (int i = 0; i < CLI_OPT_AMOUNT; i++)
221  {
222  // skip non mandatory options
223  if (!options.option[i].mandatory_option)
224  {
225  continue;
226  }
227 
228  if (!strcmp(opt, options.option[i].long_opt))
229  {
230  found_opt = 1;
231 
232  // parse option argument and set in global config
233  // do not force override, if value is already set
234  // in the global config
235  if ((ret = options.option[i].parse_option(cnf, arg, 0)) == -1)
236  {
237  error = 1;
238  break;
239  }
240 
241  // increment counter of set required options
242  // if the parsing function has set the options value
243  // in the global conf
244  if (!ret)
245  {
246  (*required_set)++;
247  }
248  }
249  }
250 
251  // raise error if read option is an unsupported option
252  if (!found_opt)
253  {
254  error = 1;
255  break;
256  }
257 
258  // increase line pointer
259  lctr++;
260  free(line);
261  }
262 
263  fclose(f);
264 
265  if (error)
266  {
267  return lctr;
268  }
269 
270  return 0;
271 }
