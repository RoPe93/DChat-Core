70 int
71 main(int argc, char** argv)
72 {
73  char option[2]; // getopt option
74  int required = 0; // counter for required options
75  int required_set = 0; // counter for set required options
76  char* short_opts = NULL; // gnu getopt string
77  struct option* long_opts; // gnu getopt long options
78  int found_opt; // boolean if opt has been found
79  cli_options_t options; // available command line options
80  char* remote_onion = NULL; // onion id of remote host
81  int rport; // remote port
82  int ret;
83 
84  if (init_global_config(&cnf) == -1)
85  {
86  fatal("Initialization of global configuration failed!");
87  }
88 
89  if (init_cli_options(&options) == -1)
90  {
91  fatal("Initialization of command line options failed!");
92  }
93 
94  short_opts = get_short_options(&options);
95  long_opts = get_long_options(&options);
96 
97  // deterime amount of required options
98  for (int i = 0; i < CLI_OPT_AMOUNT; i++)
99  {
100  if (options.option[i].mandatory_option)
101  {
102  required++;
103  }
104  }
105 
106  while (1)
107  {
108  ret = getopt_long(argc, argv, short_opts, long_opts, 0);
109 
110  // end of options
111  if (ret == -1)
112  {
113  break;
114  }
115 
116  option[0] = ret;
117  option[1] = '\0';
118  // iterate available options
119  found_opt = 0;
120 
121  for (int i = 0; i < CLI_OPT_AMOUNT; i++)
122  {
123  if (!strncmp(&options.option[i].opt, option, 1))
124  {
125  found_opt = 1;
126 
127  if ((ret = options.option[i].parse_option(&cnf, optarg, 1)) == -1)
128  {
129  usage(EXIT_FAILURE, &options, "Invalid argument '%s' for option '-%c / --%s'",
130  optarg, options.option[i].opt, options.option[i].long_opt);
131  }
132 
133  // increment counter of set required options
134  // if the parsing function has set the options value
135  // in the global conf
136  if (options.option[i].mandatory_option && !ret)
137  {
138  required_set++;
139  }
140  }
141  }
142 
143  if (!found_opt)
144  {
145  usage(EXIT_FAILURE, &options, "Invalid command-line option!");
146  }
147  }
148 
149  if (short_opts != NULL)
150  {
151  free(short_opts);
152  }
153 
154  if (long_opts != NULL)
155  {
156  free(long_opts);
157  }
158 
159  // read configuration file if existent and set corresponding
160  // values in the global config
161  if ((ret = file_exists(CONFIG_PATH)) == 1)
162  {
163  if ((ret = read_conf(&cnf, CONFIG_PATH, &required_set)) == -1)
164  {
165  log_msg(LOG_WARN, "Reading configuration file failed!");
166  }
167 
168  if (ret > 0)
169  {
170  log_msg(LOG_WARN, "Syntax error in line '%d' of config file!", ret);
171  }
172  }
173  else if (ret == -1)
174  {
175  log_errno(LOG_WARN, "Could not read configuration file '%s'!", CONFIG_PATH);
176  }
177 
178  // check if all required options have been specified
179  if (required != required_set)
180  {
181  usage(EXIT_FAILURE, &options, "Missing mandatory command-line options!");
182  }
183 
184  // client requires no arguments -> raise error if there are any
185  if (optind < argc)
186  {
187  usage(EXIT_FAILURE, &options, "Invalid command-line arguments!");
188  }
189 
190  // create listening socket
191  if (init_listening(&cnf, LISTEN_ADDR) == -1)
192  {
193  fatal("Initialization of listening socket failed!");
194  }
195 
196  if (cnf.cl.used_contacts == 1)
197  {
198  remote_onion = cnf.cl.contact[0].onion_id;
199  rport = cnf.cl.contact[0].lport;
200  }
201 
202  // init threads (connection thread, userinput thread, ...)
203  if (init_threads(&cnf) == -1)
204  {
205  fatal("Initialization of threads failed!");
206  }
207 
208  // has a remote onion address or remote port been specified? (check
209  // fake contact - see: roni_parse() / rprt_parse()) if y: connect to
210  // it
211  if (cnf.cl.used_contacts == 1)
212  {
213  // use default if onion-id has not been specified
214  if (is_valid_onion(cnf.cl.contact[0].onion_id))
215  {
216  remote_onion = cnf.cl.contact[0].onion_id;
217  }
218  else
219  {
220  remote_onion = cnf.me.onion_id;
221  }
222 
223  // use default if remote port has not been specified
224  if (is_valid_port(cnf.cl.contact[0].lport))
225  {
226  rport = cnf.cl.contact[0].lport;
227  }
228  else
229  {
230  rport = DEFAULT_PORT;
231  }
232 
233  // delete fake contact
234  del_contact(&cnf, 0);
235  // inform connection handler to connect to the specified
236  // remote host
237  write(cnf.connect_fd[1], remote_onion, ONION_ADDRLEN);
238  write(cnf.connect_fd[1], &rport, sizeof(uint16_t));
239  }
240 
241  // handle userinput
242  ret = th_new_input(&cnf);
243  // cleanup all ressources
244  destroy(&cnf);
245  return ret == 0 ? EXIT_SUCCESS : EXIT_FAILURE;
246 }

276 int
277 init_listening(dchat_conf_t* cnf, char* address)
278 {
279  int s; // local socket descriptor
280  int on = 1; // turn on a socket option
281  struct sockaddr_storage sa;
282  // setup local listening socket address
283  memset(&sa, 0, sizeof(sa));
284 
285  if (inet_pton(AF_INET, address, &((struct sockaddr_in*)&sa)->sin_addr) != 1)
286  {
287  log_msg(LOG_ERR, "Invalid listening ip address '%s'", address);
288  return -1;
289  }
290 
291  ((struct sockaddr_in*)&sa)->sin_family = AF_INET;
292 
293  if (!cnf->me.lport)
294  {
295  cnf->me.lport = DEFAULT_PORT;
296  ((struct sockaddr_in*)&sa)->sin_port = htons(DEFAULT_PORT);
297  }
298  else
299  {
300  ((struct sockaddr_in*)&sa)->sin_port = htons(cnf->me.lport);
301  }
302 
303  // create socket
304  if ((s = socket(AF_INET, SOCK_STREAM, 0)) == -1)
305  {
306  log_errno(LOG_ERR, "Creation of socket failed!");
307  return -1;
308  }
309 
310  // socketoption to prevent: 'bind() address already in use'
311  if (setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0)
312  {
313  log_errno(LOG_ERR,
314  "Setting socket options to reuse an already bound address failed!");
315  close(s);
316  return -1;
317  }
318 
319  // bind socket to a socket address
320  if (bind(s, (struct sockaddr*) &sa, sizeof(struct sockaddr)) == -1)
321  {
322  log_errno(LOG_ERR, "Binding to socket address failed!");
323  close(s);
324  return -1;
325  }
326 
327  // listen on this socket
328  if (listen(s, LISTEN_BACKLOG) == -1)
329  {
330  log_errno(LOG_ERR, "Listening on socket descriptor failed!");
331  close(s);
332  return -1;
333  }
334 
335  // set socket address, where this client is listening
336  memcpy(&cnf->sa, &sa, sizeof(struct sockaddr_storage));
337  cnf->acpt_fd = s; // set socket file descriptor
338  return s;
339 }

351 init_threads(dchat_conf_t* cnf)
352 {
353  struct sigaction sa_terminate; // signal action for program termination
354  sigset_t sigmask;
355  sigemptyset(&sigmask);
356  sigaddset(&sigmask, SIGHUP);
357  sigaddset(&sigmask, SIGQUIT);
358  sigaddset(&sigmask, SIGINT);
359  sigaddset(&sigmask, SIGTERM);
360  // all threads should block the following signals
361  pthread_sigmask(SIG_BLOCK, &sigmask, NULL);
362  sa_terminate.sa_handler = terminate;
363  // do not allow interruption of a signal handler
364  sa_terminate.sa_mask = sigmask;
365  sa_terminate.sa_flags = 0;
366  sigaction(SIGHUP, &sa_terminate, NULL); // terminal line hangup
367  sigaction(SIGQUIT, &sa_terminate, NULL); // quit programm
368  sigaction(SIGINT, &sa_terminate, NULL); // interrupt programm
369  sigaction(SIGTERM, &sa_terminate, NULL); // software termination
370 
371  // pipe to the th_new_conn
372  if (pipe(cnf->connect_fd) == -1)
373  {
374  log_errno(LOG_ERR, "Creation of connection pipe failed!");
375  return -1;
376  }
377 
378  // pipe to send signal to wait loop from connect
379  if (pipe(cnf->cl_change) == -1)
380  {
381  log_errno(LOG_ERR, "Creation of change pipe failed!");
382  return -1;
383  }
384 
385  // pipe to signal new user input from stdin
386  if (pipe(cnf->user_input) == -1)
387  {
388  log_errno(LOG_ERR, "Creation of userinput pipe faild!");
389  return -1;
390  }
391 
392  // init the mutex function used for locking the contactlist
393  if (pthread_mutex_init(&cnf->cl.cl_mx, NULL))
394  {
395  log_errno(LOG_ERR, "Initialization of mutex failed!");
396  return -1;
397  }
398 
399  // create new th_new_conn-thread
400  if (pthread_create
401  (&cnf->conn_th, NULL, (void* (*)(void*)) th_new_conn, cnf) == -1)
402  {
403  log_errno(LOG_ERR, "Creation of connection thread failed!");
404  return -1;
405  }
406 
407  // create new thread for handling userinput from stdin
408  if (pthread_create
409  (&cnf->select_th, NULL, (void* (*)(void*)) th_main_loop, cnf) == -1)
410  {
411  log_errno(LOG_ERR, "Creation of selection thread failed!");
412  return -1;
413  }
414 
415  // main thread should not block any signals
416  pthread_sigmask(SIG_UNBLOCK, &sigmask, NULL);
417  return 0;
418 }

446  close(cnf->user_input[1]);
447  // delete readline prompt and return to beginning of current line

449  dprintf(cnf->out_fd, "%s", ansi_cr());
450  dprintf(cnf->out_fd, "Good Bye!\n");

483 int
484 handle_local_input(dchat_conf_t* cnf, char* line)
485 {
486  dchat_pdu_t msg; // pdu containing the chat text message
487  int i, ret = 0, len;
488 
489  // check if user entered command
490  if ((ret = parse_cmd(cnf, line)) == 0 || ret == 1)
491  {
492  return 0;
493  }
494  // no command has been entered / or command could not be processed
495  else
496  {
497  ret = 0;
498  len += strlen(line); // memory for text message
499 
500  if (len != 0)
501  {
502  // inititialize pdu
503  if (init_dchat_pdu(&msg, CTT_ID_TXT, 1.0, cnf->me.onion_id, cnf->me.lport,
504  cnf->me.name) == -1)
505  {
506  log_msg(LOG_ERR, "Initialization of PDU failed!");
507  return -1;
508  }
509 
510  // set content of pdu
511  init_dchat_pdu_content(&msg, line, strlen(line));
512 
513  // write pdu to known contacts
514  for (i = 0; i < cnf->cl.cl_size; i++)
515  {
516  if (cnf->cl.contact[i].fd)
517  {
518  ret = write_pdu(cnf->cl.contact[i].fd, &msg);
519  }
520  }
521 
522  free_pdu(&msg);
523  }
524  }
525 
526  // return value of write_pdu
527  return ret != -1 ? 0 : -1;
528 }

539 int
540 handle_remote_input(dchat_conf_t* cnf, int n)
541 {
542  dchat_pdu_t pdu; // pdu read from contact file descriptor
543  char* txt_msg; // message used to store remote input
544  int ret; // return value
545  int len; // amount of bytes read
546  contact_t* contact; // contact to send a message to
547  contact = &cnf->cl.contact[n];
548 
549  // read pdu from file descriptor (-1 indicates error)
550  if ((len = read_pdu(contact->fd, &pdu)) == -1)
551  {
552  log_msg(LOG_ERR, "Illegal PDU from '%s'!", contact->name);
553  return -1;
554  }
555  // EOF
556  else if (!len)
557  {
558  log_msg(LOG_INFO, "'%s' disconnected!", contact->name);
559  return 0;
560  }
561 
562  // the first pdus of a newly connected client have to be a
563  // "control/discover" containing the onion-id, listening
564  // port and nickname, otherwise raise an error and delete
565  // this contact
566  if ((contact->onion_id[0] == '\0' || !contact->lport ||
567  contact->name[0] == '\0') &&
568  pdu.content_type != CTT_ID_DSC)
569  {
570  log_msg(LOG_ERR, "Client '%d' omitted identification!", n);
571  return -1;
572  }
573 
574  // check mandatory headers received
575  if (contact->name[0] != '\0' && strcmp(contact->name, pdu.nickname) != 0)
576  {
577  log_msg(LOG_INFO, "'%s' changed nickname to '%s'!", contact->name,
578  pdu.nickname);
579  }
580 
581  if (contact->onion_id[0] != '\0' &&
582  strcmp(contact->onion_id, pdu.onion_id) != 0)
583  {
584  log_msg(LOG_ERR, "'%s' changed Onion-ID! Contact will be removed!",
585  contact->name);
586  return -1;
587  }
588 
589  if (contact->lport != 0 && contact->lport != pdu.lport)
590  {
591  log_msg(LOG_ERR, "'%s' changed Listening Port! Contact will be removed!",
592  contact->name);
593  return -1;
594  }
595 
596  // set nickname of contact
597  contact->name[0] = '\0';
598 
599  if (pdu.nickname != NULL)
600  {
601  strncat(contact->name, pdu.nickname, MAX_NICKNAME);
602  }
603 
604  // set onion id of contact
605  contact->onion_id[0] = '\0';
606 
607  if (pdu.onion_id != NULL)
608  {
609  strncat(contact->onion_id, pdu.onion_id, ONION_ADDRLEN);
610  }
611 
612  // set listening port of contact
613  contact->lport = pdu.lport;
614 
615  /*
616  * == TEXT/PLAIN ==
617  */
618  if (pdu.content_type == CTT_ID_TXT)
619  {
620  // allocate memory for text message
621  if ((txt_msg = malloc(pdu.content_length + 1)) == NULL)
622  {
623  fatal("Memory allocation for text message failed!");
624  }
625 
626  // store bytes from pdu in txt_msg and terminate it
627  memcpy(txt_msg, pdu.content, pdu.content_length);
628  txt_msg[pdu.content_length] = '\0';
629  // print text message
630  print_dchat_msg(pdu.nickname, txt_msg, cnf->out_fd);
631  free(txt_msg);
632  }
633  /*
634  * == CONTROL/DISCOVER ==
635  */
636  else if (pdu.content_type == CTT_ID_DSC)
637  {
638  // since dchat brings with the problem of duplicate contacts
639  // check if there are duplicate contacts in the contactlist
640  if ((ret = check_duplicates(cnf, n)) != -1) //error
641  {
642  log_msg(LOG_INFO, "Detected duplicate contact - removing it!");
643  del_contact(cnf, ret); // delete duplicate
644  }
645 
646  // iterate through the content of the pdu containing
647  // the new contacts
648  if ((ret = receive_contacts(cnf, &pdu)) == -1)
649  {
650  log_msg(LOG_WARN, "Could not add all contacts from the received contactlist!");
651  }
652  }
653  /*
654  * == UNKNOWN CONTENT-TYPE ==
655  */
656  else
657  {
658  log_msg(LOG_WARN, "Unknown Content-Type!");
659  }
660 
661  free_pdu(&pdu);
662  return len;
663 }

778 void*
779 th_new_conn(dchat_conf_t* cnf)
780 {
781  char onion_id[ONION_ADDRLEN + 1]; // onion address of remote host
782  uint16_t port; // port of remote host
783  char c = '1'; // signal that a new connection has been established
784  int ret;
785  // setup cleanup handler and cancelation attributes
786  pthread_cleanup_push(cleanup_th_new_conn, NULL);
787  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
788  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
789 
790  for (;;)
791  {
792  // read from pipe to get new address
793  if ((ret = read(cnf->connect_fd[0], onion_id, ONION_ADDRLEN)) == -1)
794  {
795  log_msg(LOG_WARN, "Could not read Onion-ID from connection pipe!");
796  }
797 
798  // EOF
799  if (!ret)
800  {
801  break;
802  }
803 
804  // read from pipe to get new port
805  if ((ret = read(cnf->connect_fd[0], &port, sizeof(uint16_t))) == -1)
806  {
807  log_msg(LOG_WARN, "Could not read Listening-Port from connection pipe!");
808  }
809 
810  // EOF
811  if (!ret)
812  {
813  break;
814  }
815 
816  // terminate address
817  onion_id[ONION_ADDRLEN] = '\0';
818  // lock contactlist
819  pthread_mutex_lock(&cnf->cl.cl_mx);
820 
821  if (handle_local_conn_request(cnf, onion_id, port) == -1)
822  {
823  log_msg(LOG_WARN, "Connection to remote host failed!");
824  }
825  else if ((write(cnf->cl_change[1], &c, sizeof(c))) == -1)
826  {
827  log_msg(LOG_WARN, "Could not write to change pipe!");
828  }
829 
830  // unlock contactlist
831  pthread_mutex_unlock(&cnf->cl.cl_mx);
832  }
833 
834  // execute cleanup handler
835  pthread_cleanup_pop(1);
836  pthread_exit(NULL);
837 }

849 int
850 th_new_input(dchat_conf_t* cnf)

852  char prompt[MAX_NICKNAME + 8]; // readline prompt contains nickname

955 {
956  fd_set rset; // list of readable file descriptors
957  int nfds; // number of fd in rset
958  int ret; // return value
959  char c; // for pipe: th_new_conn
960  char* line; // line returned from GNU readline
961  int cancel = 0; // cancel main loop
962  int i;

964  pthread_cleanup_push(cleanup_th_main_loop, NULL);
965  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
966  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
967 
968  for (;;)
969  {
970  // INIT fd_set
971  FD_ZERO(&rset);
972  // ADD STDIN: pipe file descriptor that connects the thread
973  // function 'th_new_input'
974  FD_SET(cnf->user_input[0], &rset);
975  // ADD LISTENING PORT: acpt_fd of global config
976  FD_SET(cnf->acpt_fd, &rset);
977  nfds = max(cnf->user_input[0], cnf->acpt_fd);
978  // ADD NEW CONN: pipe file descriptor that connects the thead
979  // function 'th_new_conn'
980  FD_SET(cnf->cl_change[0], &rset);
981  nfds = max(nfds, cnf->cl_change[0]);
982  // ADD CONTACTS: add all contact socket file descriptors from
983  // the contactlist
984  pthread_mutex_lock(&cnf->cl.cl_mx);
985 
986  for (i = 0; i < cnf->cl.cl_size; i++)
987  {
988  if (cnf->cl.contact[i].fd)
989  {
990  FD_SET(cnf->cl.contact[i].fd, &rset);
991  // determine fd with highest value
992  nfds = max(cnf->cl.contact[i].fd, nfds);
993  }
994  }
995 
996  pthread_mutex_unlock(&cnf->cl.cl_mx);
997  // used as backup since nfds will be overwritten if an
998  // interrupt
999  // occurs
1000  int old_nfds = nfds;
1001  pthread_testcancel();
1002 
1003  while ((nfds = select(old_nfds + 1, &rset, NULL, NULL, NULL)) <= 0)
1004  {
1005  pthread_testcancel();
1006 
1007  if (nfds == -1)
1008  {
1009  // something interrupted select(2) - try select again
1010  if (errno == EINTR)
1011  {
1012  continue;
1013  }
1014 
1015  log_errno(LOG_ERR, "select() failed!");
1016  cancel = 1;
1017  break;
1018  }
1019  }
1020 
1021  if (cancel)
1022  {
1023  break;
1024  }
1025 
1026  // CHECK STDIN: check if thread has written to the user_input
1027  // pipe
1028  if (FD_ISSET(cnf->user_input[0], &rset))
1029  {
1030  nfds--;
1031 
1032  // read length of string from pipe
1033  if (read(cnf->user_input[0], &ret, sizeof(int)) < 0)
1034  {
1035  break;
1036  }
1037 
1038  // allocate memory for the string entered from user
1039  line = malloc(ret + 1);
1040 
1041  // read string
1042  if (read(cnf->user_input[0], line, ret) < 0 || ret <= 0)
1043  {
1044  free(line);
1045  break;
1046  }
1047 
1048  line[ret] = '\0';
1049  pthread_mutex_lock(&cnf->cl.cl_mx);
1050 
1051  // handle user input
1052  if ((ret = handle_local_input(cnf, line)) == -1)
1053  {
1054  break;
1055  }
1056 
1057  pthread_mutex_unlock(&cnf->cl.cl_mx);
1058  free(line);
1059  }
1060

1062  // accepted
1063  if (FD_ISSET(cnf->acpt_fd, &rset))
1064  {
1065  nfds--;
1066  pthread_mutex_lock(&cnf->cl.cl_mx);
1067 
1068  // handle new connection request
1069  if ((ret = handle_remote_conn_request(cnf)) == -1)
1070  {
1071  break;
1072  }
1073 
1074  pthread_mutex_unlock(&cnf->cl.cl_mx);
1075  }
1076 
1077  // CHECK NEW CONN: check if user new connection has been added
1078  if (FD_ISSET(cnf->cl_change[0], &rset))
1079  {
1080  nfds--;
1081 
1082  // !< EOF
1083  if (read(cnf->cl_change[0], &c, sizeof(c)) < 0)
1084  {
1085  break;
1086  }
1087  }
1088 
1089  // CHECK CONTACTS: check file descriptors of contacts
1090  // check if nfds is 0 => no contacts have written something
1091  pthread_mutex_lock(&cnf->cl.cl_mx);
1092 
1093  for (i = 0; nfds && i < cnf->cl.cl_size; i++)
1094  {
1095  if (FD_ISSET(cnf->cl.contact[i].fd, &rset))
1096  {
1097  nfds--;
1098

1100  // -1 = error, 0 = EOF
1101  if ((ret = handle_remote_input(cnf, i)) == -1 || ret == 0)
1102  {
1103  del_contact(cnf, i);
1104  }
1105  }
1106  }
1107

1109  }
1110

1112  pthread_cleanup_pop(1);
1113  pthread_exit(NULL);
1114 }
