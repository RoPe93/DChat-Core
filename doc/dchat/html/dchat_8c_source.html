72 main(int argc, char** argv)
73 {
74  char* local_onion = NULL; // local onion address
75  int lport = DEFAULT_PORT; // local listening port
76  char* nickname = NULL; // nickname to use within chat
77  char* remote_onion = NULL; // onion address of remote client
78  int rport = -1; // remote port of remote host
79  int option; // getopt option
80  int required = 0; // counter for required options
81  int found_int = 0; // check existence of network int
82  int ret;
83  char* term; // used for strtol
84  struct sockaddr_storage sa; // local socket address
85  struct sockaddr_storage tor_client; // socket address of tor client
86  //options for execution in terminal
87  struct option long_options[] =
88  {
89  {"lonion", required_argument, 0, 's'},
90  {"nickname", required_argument, 0, 'n'},
91  {"lport", required_argument, 0, 'l'},
92  {"ronion", required_argument, 0, 'd'},
93  {"rport", required_argument, 0, 'r'}
94  };
95 
96  // parse commandline options
97  while (1)
98  {
99  option = getopt_long(argc, argv, "s:n:l:d:r:", long_options, 0);
100 
101  // end of options
102  if (option == -1)
103  {
104  break;
105  }
106 
107  switch (option)
108  {
109  case 's':
110  local_onion = optarg;
111  required++;
112  break;
113 
114  case 'n':
115  nickname = optarg;
116  required++;
117  break;
118 
119  case 'l':
120  lport = (int) strtol(optarg, &term, 10);
121 
122  if (lport < 0 || lport > 65535 || *term != '\0')
123  {
124  log_msg(LOG_ERR, "Invalid listening port '%s'", optarg);
125  usage();

128  break;
129 
130  case 'd':
131  remote_onion = optarg;
132  break;
133 
134  case 'r':
135  rport = (int) strtol(optarg, &term, 10);
136 
137  if (rport < 0 || rport > 65535 || *term != '\0')
138  {
139  log_msg(LOG_ERR, "Invalid remote port '%s'", optarg);
140  usage();
141  return EXIT_FAILURE;
142  }
143  break;
144 
145  // invalid option - getopt prints error msg
146  case '?':
147  default:
148  usage();
149  return EXIT_FAILURE;
150  }
151  }
152 
153  //FIXME: check if onion address is valid
154  // local onion address and nick are mandatory; only options arguments are allowed
155  if (required < 2 || optind < argc)
156  {
157  usage();
158  return EXIT_FAILURE;
159  }
160 
161  // local listening socket address
162  memset(&sa, 0, sizeof(sa));
163  if(inet_pton(AF_INET, LISTEN_ADDR, &((struct sockaddr_in*)&sa)->sin_addr) != 1)

167  }

169  ((struct sockaddr_in*)&sa)->sin_port = htons(lport);

172  if (init(&cnf, &sa, local_onion, nickname) == -1)
173  {
174  return EXIT_FAILURE;
175  }
176 
177  // has a remote onion address or remote port been specified? if y: connect to it
178  if (remote_onion != NULL || rport != -1)
179  {
180  if(remote_onion == NULL){
181  remote_onion = local_onion;
182  }
183  if(rport == -1){
184  rport = DEFAULT_PORT;
185  }
186 
187  pthread_mutex_lock(&cnf.cl.cl_mx);
188  // connect to remote client and send contactlist
189  if (handle_local_conn_request(&cnf, remote_onion, rport) == -1)
190  {
191  log_msg(LOG_DEBUG, "main(): Connection to remote host failed!");
192  }
193  pthread_mutex_unlock(&cnf.cl.cl_mx);
194  }
195 
196  // main chat loop
197  ret = th_new_input(&cnf);
198  // cleanup all ressources
199  destroy(&cnf);
200  return ret == 0 ? EXIT_SUCCESS : EXIT_FAILURE;
201 }

217 init(dchat_conf_t* cnf, struct sockaddr_storage* sa, char* onion_id, char* nickname)
218 {
219  struct sigaction sa_terminate; // signal action for program termination
220  sigset_t sigmask;
221  sigemptyset(&sigmask);
222  sigaddset(&sigmask, SIGHUP);
223  sigaddset(&sigmask, SIGQUIT);
224  sigaddset(&sigmask, SIGINT);
225  sigaddset(&sigmask, SIGTERM);
226  // all threads should block the following signals
227  pthread_sigmask(SIG_BLOCK, &sigmask, NULL);
228  sa_terminate.sa_handler = terminate;
229  // do not allow interruption of a signal handler
230  sa_terminate.sa_mask = sigmask;
231  sa_terminate.sa_flags = 0;
232  sigaction(SIGHUP, &sa_terminate, NULL); // terminal line hangup
233  sigaction(SIGQUIT, &sa_terminate, NULL); // quit programm
234  sigaction(SIGINT, &sa_terminate, NULL); // interrupt programm
235  sigaction(SIGTERM, &sa_terminate, NULL); // software termination
236 
237  // init the listening socket
238  if (init_global_config(cnf, sa, onion_id, nickname) == -1)
239  {
240  return -1;
241  }
242 
243  // pipe to the th_new_conn
244  if (pipe(cnf->connect_fd) == -1)
245  {
246  log_errno(LOG_ERR, "could not create pipe");
247  return -1;
248  }
249 
250  // pipe to send signal to wait loop from connect
251  if (pipe(cnf->cl_change) == -1)
252  {
253  log_errno(LOG_ERR, "could not create cl_change pipe");
254  return -1;
255  }
256 
257  // pipe to signal new user input from stdin
258  if (pipe(cnf->user_input) == -1)
259  {
260  log_errno(LOG_ERR, "could not create cl_change pipe");
261  return -1;
262  }
263 
264  // init the mutex function used for locking the contactlist
265  if (pthread_mutex_init(&cnf->cl.cl_mx, NULL))
266  {
267  log_errno(LOG_ERR, "pthread_mutex_init() failed");
268  return -1;
269  }
270 
271  // create new th_new_conn-thread
272  if (pthread_create
273  (&cnf->conn_th, NULL, (void* (*)(void*)) th_new_conn, cnf) == -1)
274  {
275  log_errno(LOG_ERR, "pthread_create() failed");
276  return -1;
277  }
278 
279  // create new thread for handling userinput from stdin
280  if (pthread_create
281  (&cnf->select_th, NULL, (void* (*)(void*)) th_main_loop, cnf) == -1)
282  {
283  log_errno(LOG_ERR, "pthread_create() failed");
284  return -1;
285  }
286 
287  // main thread should not block any signals
288  pthread_sigmask(SIG_UNBLOCK, &sigmask, NULL);
289  return 0;
290 }

305 init_global_config(dchat_conf_t* cnf, struct sockaddr_storage* sa, char* onion_id,
306  char* nickname)
307 {
308  char addr_str[INET6_ADDRSTRLEN + 1]; // ip as string
309  int s; // local socket descriptor
310  int on = 1; // turn on a socket option
311  int found_int = 0; // check if interface was found
312 
313  // create socket
314  if ((s = socket(AF_INET, SOCK_STREAM, 0)) == -1)
315  {
316  log_errno(LOG_ERR, "socket() failed");
317  return -1;
318  }
319 
320  // socketoption to prevent: 'bind() address already in use'
321  if (setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0)
322  {
323  log_errno(LOG_ERR, "setsockopt() failed");
324  close(s);
325  return -1;
326  }
327 
328  // bind socket to a socket address
329  if (bind(s, (struct sockaddr*) sa, sizeof(struct sockaddr)) == -1)
330  {
331  log_errno(LOG_ERR, "bind() failed");
332  close(s);
333  return -1;
334  }
335 
336  // listen on this socket
337  if (listen(s, LISTEN_BACKLOG) == -1)
338  {
339  log_errno(LOG_ERR, "listen() failed");
340  close(s);
341  return -1;
342  }
343 
344  // init the global config structure
345  memset(cnf, 0, sizeof(*cnf));
346  cnf->in_fd = 0; // use stdin as input source
347  cnf->out_fd = 1; // use stdout as output source
348  cnf->cl.cl_size = 0; // set initial size of contactlist
349  cnf->cl.used_contacts = 0; // no known contacts, at start
350  cnf->me.name[0] = '\0';
351  strncat(cnf->me.name, nickname, MAX_NICKNAME); // set nickname
352  cnf->me.onion_id[0] = '\0';
353  strncat(cnf->me.onion_id, onion_id, ONION_ADDRLEN); // set onion address
354  // listening port
355  if (ip_version(sa) == 4)
356  {
357  log_msg(LOG_INFO, "Listening on '%s:%d'", onion_id, ntohs(((struct sockaddr_in*) sa)->sin_port));
358  cnf->me.lport = ntohs(((struct sockaddr_in*) sa)->sin_port);
359  }
360  else if (ip_version(sa) == 6)
361  {
362  log_msg(LOG_INFO, "Listening on '%s:%d'", onion_id, ntohs(((struct sockaddr_in6*) sa)->sin6_port));
363  cnf->me.lport = ntohs(((struct sockaddr_in6*) sa)->sin6_port);
364  }
365  else
366  {
367  log_msg(LOG_ERR, "Invalid socket address!");
368  close(s);
369  return -1;
370  }
371  // set socket address, where this client is listening
372  memcpy(&cnf->sa, sa, sizeof(struct sockaddr_storage));
373  cnf->acpt_fd = s; // set socket file descriptor
374  return s;
375 }

437 int
438 handle_local_input(dchat_conf_t* cnf, char* line)
439 {
440  dchat_pdu_t msg; // pdu containing the chat text message
441  int i, ret = 0, len, cmd;
442 
443  // user entered command
444  if ((cmd = parse_cmd(cnf, line)) == 0)
445  {
446  return 0;
447  }
448  // command has wrong syntax
449  else if (cmd == 1)
450  {
451  log_msg(LOG_WARN, "Syntax error");
452  return -2;
453  }
454  // no command has been entered
455  else
456  {
457  len += strlen(line); // memory for text message
458 
459  if (len != 0)
460  {
461  // init dchat pdu
462  memset(&msg, 0, sizeof(msg));
463  msg.content_type = CT_TXT_PLAIN;
464  msg.onion_id = cnf->me.onion_id;
465  msg.lport = cnf->me.lport;
466  msg.nickname = cnf->me.name;
467 
468  // allocate memory the size of nickname + 2 + text message
469  if ((msg.content = malloc(len + 1)) == NULL)
470  {
471  log_errno(LOG_ERR,
472  "handle_local_input() failed, could not allocated memory");
473  return -1;
474  }
475 
476  // append the content
477  msg.content[0] = '\0';
478  strncat(msg.content, line, strlen(line));
479  msg.content_length = len; // length of content excluding \0
480 
481  // write pdu to known contacts
482  for (i = 0; i < cnf->cl.cl_size; i++)
483  {
484  if (cnf->cl.contact[i].fd)
485  {
486  ret = write_pdu(cnf->cl.contact[i].fd, &msg);
487  }
488  }
489 
490  free(msg.content);
491  }
492  else
493  {
494  log_msg(LOG_WARN, "Did not send empty string!");
495  }
496  }
497 
498  // return value of write_pdu
499  return ret != -1 ? 0 : -1;
500 }

511 int
512 handle_remote_input(dchat_conf_t* cnf, int n)
513 {
514  dchat_pdu_t* pdu; // pdu read from contact file descriptor
515  char* txt_msg; // message used to store remote input
516  int ret; // return value
517  int len; // amount of bytes read
518  int fd; // file descriptor of a contact
519  fd = cnf->cl.contact[n].fd;
520 
521  // read pdu from file descriptor (-1 indicates error)
522  if ((len = read_pdu(fd, &pdu)) == -1)
523  {
524  log_msg(LOG_ERR, "Illegal PDU from contact %d", n);
525  return -1;
526  }
527  // EOF
528  else if (!len)
529  {
530  log_msg(LOG_INFO, "contact '%d' disconnected", n);
531  return 0;
532  }
533 
534  // the first pdu of a newly connected client has to be a
535  // "control/discover" containing the onion-id and the listening
536  // port otherwise raise an error and delete
537  // this contact
538  if ((cnf->cl.contact[n].onion_id[0] == '\0' || !cnf->cl.contact[n].lport) && pdu->content_type != CT_CTRL_DISC)
539  {
540  log_msg(LOG_ERR, "Contact '%s' has not identfied himself", cnf->cl.contact[n].name);
541  return -1;
542  }
543  else if (pdu->content_type == CT_CTRL_DISC)
544  {
545  if(cnf->cl.contact[n].name[0] == '\0'){
546  // set nickname of contact
547  strncat(cnf->cl.contact[n].name, pdu->nickname, MAX_NICKNAME);
548  }
549  if(cnf->cl.contact[n].onion_id[0] == '\0'){
550  // set onion id of contact
551  strncat(cnf->cl.contact[n].onion_id, pdu->onion_id, ONION_ADDRLEN);
552  }
553  if(!cnf->cl.contact[n].lport){
554  // set listening port of contact
555  cnf->cl.contact[n].lport = pdu->lport;
556  }
557  }
558 
559  /*
560  * == TEXT/PLAIN ==
561  */
562  if (pdu->content_type == CT_TXT_PLAIN)
563  {
564  // allocate memory for text message
565  if ((txt_msg = malloc(pdu->content_length + 1)) == NULL)
566  {
567  log_errno(LOG_ERR,
568  "handle_remote_input() failed - Could not allocate memory");
569  free_pdu(pdu);
570  return -1;
571  }
572 
573  // store bytes from pdu in txt_msg and terminate it
574  memcpy(txt_msg, pdu->content, pdu->content_length);
575  txt_msg[pdu->content_length] = '\0';
576  // print text message
577  print_dchat_msg(pdu->nickname, txt_msg, cnf->out_fd);
578  free(txt_msg);
579  }
580  /*
581  * == CONTROL/DISCOVER ==
582  */
583  else if (pdu->content_type == CT_CTRL_DISC)
584  {
585  // since dchat brings with the problem of duplicate contacts
586  // check if there are duplicate contacts in the contactlist
587  if ((ret = check_duplicates(cnf, n)) == -2)
588  {
589  // ERROR
590  log_msg(LOG_ERR, "check_duplicate_contact() failed");
591  }
592  // no duplicates
593  else if (ret == -1)
594  {
595  // log_msg(LOG_INFO, "no duplicates founds");
596  }
597  // duplicates found
598  else
599  {
600  log_msg(LOG_INFO, "Duplicate detected - removing it (%d)", ret);
601  del_contact(cnf, ret); // delete duplicate
602  }
603 
604  // iterate through the content of the pdu containing
605  // the new contacts
606  if ((ret = receive_contacts(cnf, pdu)) == -1)
607  {
608  log_msg(LOG_WARN, "Not every new contact could be added to contactlist");
609  }
610  else if (ret == 0)
611  {
612  // log_msg(LOG_DEBUG, "client %d does not know any other
613  // contacts we do not know about", n);
614  }
615  else
616  {
617  // log_msg(LOG_DEBUG, "%d new client(s) have been added",
618  // ret);
619  }
620  }
621 
622  /*
623  * == UNKNOWN CONTENT-TYPE ==
624  */
625  else
626  {
627  log_msg(LOG_WARN, "Unknown Content-Type!");
628  }
629 
630  free_pdu(pdu);
631  return len;
632 }

753 th_new_conn(dchat_conf_t* cnf)
754 {
755  char onion_id[ONION_ADDRLEN + 1]; // destination address to connect to
756  uint16_t port;
757  char c = '1'; // will be written to `cl_change`
758  int ret;
759  // setup cleanup handler and cancelation attributes
760  pthread_cleanup_push(cleanup_th_new_conn, NULL);
761  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
762  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
763 
764  for (;;)
765  {
766  // read from pipe to get new address
767  if ((ret = read(cnf->connect_fd[0], &onion_id, ONION_ADDRLEN)) == -1)
768  {
769  log_msg(LOG_WARN, "th_new_conn(): Could not read from pipe");
770  }
771  // EOF
772  if (!ret)
773  {
774  break;
775  }
776  // read from pipe to get new port
777  if ((ret = read(cnf->connect_fd[0], &port, sizeof(port))) == -1)
778  {
779  log_msg(LOG_WARN, "th_new_conn(): Could not read from pipe");
780  }
781  // EOF
782  if (!ret)
783  {
784  break;
785  }
786 
787  // terminate address
788  onion_id[ONION_ADDRLEN] = '\0';
789 
790  // lock contactlist
791  pthread_mutex_lock(&cnf->cl.cl_mx);
792 
793  if (handle_local_conn_request(cnf, onion_id, port) == -1)
794  {
795  log_errno(LOG_WARN, "th_new_conn(): Connection to remote host failed! ");
796  }
797  else if ((write(cnf->cl_change[1], &c, sizeof(c))) == -1)
798  {
799  log_msg(LOG_WARN, "th_new_conn(): Could not write to pipe");
800  }
801 
802  // unlock contactlist
803  pthread_mutex_unlock(&cnf->cl.cl_mx);
804  }
805 
806  // execute cleanup handler
807  pthread_cleanup_pop(1);
808  pthread_exit(NULL);
809 }

926 th_main_loop(dchat_conf_t* cnf)
927 {
928  fd_set rset; // list of readable file descriptors
929  int nfds; // number of fd in rset
930  int ret; // return value
931  char c; // for pipe: th_new_conn
932  char* line; // line returned from GNU readline
933  int i;
934  // setup cleanup handler and cancelation attributes
935  pthread_cleanup_push(cleanup_th_main_loop, NULL);
936  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
937  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
938 
939  for (;;)
940  {
941  // INIT fd_set
942  FD_ZERO(&rset);
943  // ADD STDIN: pipe file descriptor that connects the thread
944  // function 'th_new_input'
945  FD_SET(cnf->user_input[0], &rset);
946  // ADD LISTENING PORT: acpt_fd of global config
947  FD_SET(cnf->acpt_fd, &rset);
948  nfds = max(cnf->user_input[0], cnf->acpt_fd);
949  // ADD NEW CONN: pipe file descriptor that connects the thead
950  // function 'th_new_conn'
951  FD_SET(cnf->cl_change[0], &rset);
952  nfds = max(nfds, cnf->cl_change[0]);
953  // ADD CONTACTS: add all contact socket file descriptors from
954  // the contactlist
955  pthread_mutex_lock(&cnf->cl.cl_mx);
956 
957  for (i = 0; i < cnf->cl.cl_size; i++)
958  {
959  if (cnf->cl.contact[i].fd)
960  {
961  FD_SET(cnf->cl.contact[i].fd, &rset);
962  // determine fd with highest value
963  nfds = max(cnf->cl.contact[i].fd, nfds);
964  }
965  }
966 
967  pthread_mutex_unlock(&cnf->cl.cl_mx);
968  // used as backup since nfds will be overwritten if an
969  // interrupt
970  // occurs
971  int old_nfds = nfds;
972 
973  while ((nfds = select(old_nfds + 1, &rset, NULL, NULL, NULL)) == -1)
974  {
975  // something interrupted select(2) - try select again
976  if (errno == EINTR)
977  {
978  continue;
979  }
980  else
981  {
982  log_errno(LOG_ERR, "select() failed");
983  //return -1;
984  break;
985  }
986  }
987 
988  // CHECK STDIN: check if thread has written to the user_input
989  // pipe
990  if (FD_ISSET(cnf->user_input[0], &rset))
991  {
992  nfds--;
993 
994  // read length of string from pipe
995  if (read(cnf->user_input[0], &ret, sizeof(int)) < 0)
996  {
997  break;
998  }
999 
1000  // allocate memory for the string entered from user
1001  line = malloc(ret + 1);
1002 
1003  // read string
1004  if (read(cnf->user_input[0], line, ret) < 0 || ret <= 0)
1005  {
1006  free(line);
1007  break;
1008  }
1009 
1010  line[ret] = '\0';
1011  pthread_mutex_lock(&cnf->cl.cl_mx);
1012 
1013  // handle user input
1014  if ((ret = handle_local_input(cnf, line)) == -1)
1015  {
1016  break;
1017  }
1018 
1019  pthread_mutex_unlock(&cnf->cl.cl_mx);
1020  free(line);
1021  }
1022 
1023  // CHECK LISTENING PORT: check if new connection can be
1024  // accepted
1025  if (FD_ISSET(cnf->acpt_fd, &rset))
1026  {
1027  nfds--;
1028  pthread_mutex_lock(&cnf->cl.cl_mx);
1029 
1030  // handle new connection request
1031  if ((ret = handle_remote_conn_request(cnf)) == -1)
1032  {
1033  break;
1034  }
1035 
1036  pthread_mutex_unlock(&cnf->cl.cl_mx);
1037  }
1038 
1039  // CHECK NEW CONN: check if user new connection has been added
1040  if (FD_ISSET(cnf->cl_change[0], &rset))
1041  {
1042  nfds--;
1043 
1044  // !< EOF
1045  if (read(cnf->cl_change[0], &c, sizeof(c)) < 0)
1046  {
1047  break;
1048  }
1049  }
1050 
1051  // CHECK CONTACTS: check file descriptors of contacts
1052  // check if nfds is 0 => no contacts have written something
1053  pthread_mutex_lock(&cnf->cl.cl_mx);
1054 
1055  for (i = 0; nfds && i < cnf->cl.cl_size; i++)
1056  {
1057  if (FD_ISSET(cnf->cl.contact[i].fd, &rset))
1058  {
1059  nfds--;
1060 
1061  // handle input from remote user
1062  // -1 = error, 0 = EOF
1063  if ((ret = handle_remote_input(cnf, i)) == -1 || ret == 0)
1064  {
1065  close(cnf->cl.contact[i].fd);
1066  del_contact(cnf, i);
1067  }
1068  }
1069  }
1070 
1071  pthread_mutex_unlock(&cnf->cl.cl_mx);
1072  }
1073 
1074  //execute cleanup handler
1075  pthread_cleanup_pop(1);
1076  pthread_exit(NULL);
1077 }
