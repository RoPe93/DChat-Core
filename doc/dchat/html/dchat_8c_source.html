70 int
71 main(int argc, char** argv)
72 {
73  int acpt_port; // local listening port
74  int remote_port; // port of remote client
75  char* interface = NULL; // interface to bind to
76  char* remote_address = NULL; // ip of remote client
77  char* nickname = NULL; // nickname to use within chat
78  //FIXME: use sockaddr_storage for IP6 support
79  struct sockaddr_in da; // socket address for remote client
80 
81  // see usage
82  if (argc < 4 || argc == 5 || argc > 6)
83  {
84  usage();
85  return EXIT_FAILURE;
86  }
87  else
88  {
89  interface = argv[1];
90  acpt_port = atoi(argv[2]);
91  nickname = argv[3];
92  }
93 
94  // init dchat (e.g. global configuration, listening socket, threads, ...)
95  if (init(&cnf, interface, acpt_port, nickname) == -1)
96  {
97  return EXIT_FAILURE;
98  }
99 
100  // has a remoteaddress and a remoteport been specified?
101  if (argc == 6)
102  {
103  remote_address = argv[4];
104  remote_port = atoi(argv[5]);
105  // init socket address structure
106  memset(&da, 0, sizeof(da));
107  da.sin_family = AF_INET;
108  da.sin_port = htons(remote_port);
109 
110  // init ip address in socket address structure
111  if (inet_pton(AF_INET, remote_address, &da.sin_addr) != 1)
112  {
113  log_errno(LOG_ERR, "wrong format of ip address");
114  }
115 
116  // connect to remote client and send contactlist
117  if (handle_local_conn_request(&cnf, (struct sockaddr_storage*) &da) ==
118  -1)
119  {
120  log_errno(LOG_WARN,
121  "main(): Could not execute connection request successfully");
122  }
123  }
124 
125  // main chat loop
126  ret_exit_ = th_main_loop(&cnf);
127  // free resources and terminate this process
128  raise(SIGTERM);
129  return EXIT_SUCCESS;
130 }

145 int
146 init(dchat_conf_t* cnf, char* interface, int acpt_port, char* nickname)
147 {
148  struct sigaction sa_terminate; // signal action for program termination
149  sa_terminate.sa_handler = terminate;
150  sigemptyset(&sa_terminate.sa_mask);
151  sa_terminate.sa_flags = 0;
152  sigaction(SIGHUP, &sa_terminate, NULL); // terminal line hangup
153  sigaction(SIGQUIT, &sa_terminate, NULL); // quit programm
154  sigaction(SIGINT, &sa_terminate, NULL); // interrupt programm
155  sigaction(SIGTERM, &sa_terminate, NULL); // software termination
156 
157  // check if given port is valid
158  if (acpt_port < 1 || acpt_port > 65535)
159  {
160  log_msg(LOG_WARN, "Invalid port for listening socket");
161  return -1;
162  }
163 
164  // init the listening socket
165  if (init_global_config(cnf, interface, acpt_port, nickname) == -1)
166  {
167  return -1;
168  }
169 
170  // pipe to the th_new_conn
171  if (pipe(cnf->connect_fd) == -1)
172  {
173  log_errno(LOG_ERR, "could not create pipe");
174  return -1;
175  }
176 
177  // pipe to send signal to wait loop from connect
178  if (pipe(cnf->cl_change) == -1)
179  {
180  log_errno(LOG_ERR, "could not create cl_change pipe");
181  return -1;
182  }
183 
184  // pipe to signal new user input from stdin
185  if (pipe(cnf->user_input) == -1)
186  {
187  log_errno(LOG_ERR, "could not create cl_change pipe");
188  return -1;
189  }
190 
191  // init the mutex function used for locking the contactlist
192  if (pthread_mutex_init(&cnf->cl.cl_mx, NULL))
193  {
194  log_errno(LOG_ERR, "pthread_mutex_init() failed");
195  return -1;
196  }
197 
198  // create new th_new_conn-thread
199  if (pthread_create
200  (&cnf->conn_th, NULL, (void* (*)(void*)) th_new_conn, cnf) == -1)
201  {
202  log_errno(LOG_ERR, "pthread_create() failed");
203  return -1;
204  }
205 
206  // create new thread for handling userinput from stdin
207  if (pthread_create
208  (&cnf->userin_th, NULL, (void* (*)(void*)) th_new_input, cnf) == -1)
209  {
210  log_errno(LOG_ERR, "pthread_create() failed");
211  return -1;
212  }
213 
214  return 0;
215 }

229 int
230 init_global_config(dchat_conf_t* cnf, char* interface, int acpt_port,
231  char* nickname)
232 {
233  struct ifaddrs* ifaddr; // information of all network interfaces
234  struct ifaddrs* ifa; // interface pointer
235  struct sockaddr sa; // socket address for local client
236  char addr_str[INET6_ADDRSTRLEN + 1]; // ip as string
237  int s; // local socket descriptor
238  int on = 1; // turn on a socket option
239  int found_int = 0; // check if interface was found
240 
241  // create socket
242  if ((s = socket(AF_INET, SOCK_STREAM, 0)) == -1)
243  {
244  log_errno(LOG_ERR, "socket() failed");
245  return -1;
246  }
247 
248  // socketoption to prevent: 'bind() address already in use'
249  if (setsockopt(s, SOL_SOCKET, SO_REUSEADDR, &on, sizeof(on)) < 0)
250  {
251  log_errno(LOG_ERR, "setsockopt() failed");
252  close(s);
253  return -1;
254  }
255 
256  // get information about all network interfaces
257  if (getifaddrs(&ifaddr) == -1)
258  {
259  log_errno(LOG_ERR, "getifaddrs() failed");
260  close(s);
261  return -1;
262  }
263 
264  // iterate through network interfaces and set ip of interface
265  memset(&sa, 0, sizeof(sa));
266 
267  for (ifa = ifaddr; ifa != NULL; ifa = ifa->ifa_next)
268  {
269  // FIXME: Support for IPv6
270  if (ifa->ifa_addr != NULL && ifa->ifa_addr->sa_family == AF_INET)
271  {
272  // is it the interface we are looking for?
273  if (!strncmp(interface, ifa->ifa_name, strlen(ifa->ifa_name)))
274  {
275  if (ifa->ifa_addr != NULL)
276  {
277  // temporarily store address
278  memcpy(&sa, ifa->ifa_addr, sizeof(struct sockaddr));
279  found_int = 1; // we found the given interface
280  break;
281  }
282  else
283  {
284  log_msg(LOG_ERR,
285  "Interface '%s' does not have an ip address bound to it",
286  ifa->ifa_name);
287  close(s);
288  freeifaddrs(ifaddr);
289  return -1;
290  }
291  }
292  }
293  }
294 
295  // interface has not been found
296  if (!found_int)
297  {
298  log_msg(LOG_ERR, "Interface '%s' not found", interface);
299  close(s);
300  freeifaddrs(ifaddr);
301  return -1;
302  }
303 
304  freeifaddrs(ifaddr);
305 
306  // add port to socketaddress structure
307  if (ip_version((struct sockaddr_storage*) &sa) == 4)
308  {
309  ((struct sockaddr_in*) &sa)->sin_port = htons(acpt_port);
310  }
311  else if (ip_version((struct sockaddr_storage*) &sa) == 6)
312  {
313  ((struct sockaddr_in6*) &sa)->sin6_port = htons(acpt_port);
314  }
315 
316  // bind socket to a specified interface e.g. "eth0"
317  if (bind(s, &sa, sizeof(struct sockaddr)) == -1)
318  {
319  log_errno(LOG_ERR, "bind() failed");
320  close(s);
321  return -1;
322  }
323 
324  // listen on this socket
325  if (listen(s, 5) == -1)
326  {
327  log_errno(LOG_ERR, "listen() failed");
328  close(s);
329  return -1;
330  }
331 
332  // log, where on which address and port this client will be
333  // listening
334  if (ip_version((struct sockaddr_storage*) &sa) == 4)
335  {
336  log_msg(LOG_INFO, "Listening on '%s:%d'",
337  inet_ntop(AF_INET, &((struct sockaddr_in*) &sa)->sin_addr,
338  addr_str, INET_ADDRSTRLEN), acpt_port);
339  }
340  else if (ip_version((struct sockaddr_storage*) &sa) == 6)
341  {
342  log_msg(LOG_INFO, "Listening on '%s:%d'",
343  inet_ntop(AF_INET6, &((struct sockaddr_in6*) &sa)->sin6_addr,
344  addr_str, INET6_ADDRSTRLEN), acpt_port);
345  }
346 
347  // init the global config structure
348  memset(cnf, 0, sizeof(*cnf));
349  cnf->in_fd = 0; // use stdin as input source
350  cnf->out_fd = 1; // use stdout as output source
351  cnf->cl.cl_size = 0; // set initial size of contactlist
352  cnf->cl.used_contacts = 0; // no known contacts, at start
353  cnf->me.name[0] = '\0';
354  strncat(cnf->me.name, nickname, MAX_NICKNAME); // set nickname
355  cnf->me.lport = acpt_port; // set listening-port
356  // set socket address, where this client is listening
357  memcpy(&cnf->me.stor, &sa, sizeof(struct sockaddr_storage));
358  cnf->acpt_fd = s; // set socket file descriptor
359  return s;
360 }

431 int
432 handle_local_input(dchat_conf_t* cnf, char* line)
433 {
434  dchat_pdu_t msg; // pdu containing the chat text message
435  int i, ret = 0, len, cmd;
436 
437  // user entered command
438  if ((cmd = parse_cmd(cnf, line)) == 0)
439  {
440  return 0;
441  }
442  // command has wrong syntax
443  else if (cmd == -2)
444  {
445  log_msg(LOG_WARN, "Syntax error");
446  return -2;
447  }
448  // no command has been entered
449  else
450  {
451  len = strlen(cnf->me.name) + 2; // memory for local nickname +2 for ": "
452  len += strlen(line); // memory for text message
453 
454  if (len != 0)
455  {
456  // init dchat pdu
457  memset(&msg, 0, sizeof(msg));
458  msg.content_type = CT_TXT_PLAIN;
459  msg.listen_port = cnf->me.lport;
460 
461  // allocate memory the size of nickname + 2 + text message
462  if ((msg.content = malloc(len + 1)) == NULL)
463  {
464  log_errno(LOG_ERR,
465  "handle_local_input() failed, could not allocated memory");
466  return -1;
467  }
468 
469  // append the content
470  msg.content[0] = '\0';
471  strncat(msg.content, cnf->me.name, strlen(cnf->me.name));
472  strncat(msg.content, ": ", 2);
473  strncat(msg.content, line, strlen(line));
474  msg.content_length = len; // length of content excluding \0
475 
476  // write pdu to known contacts
477  for (i = 0; i < cnf->cl.cl_size; i++)
478  {
479  if (cnf->cl.contact[i].fd)
480  {
481  ret = write_pdu(cnf->cl.contact[i].fd, &msg);
482  }
483  }
484 
485  free(msg.content);
486  }
487  else
488  {
489  log_msg(LOG_WARN, "handle_local_input got empty string");
490  }
491  }
492 
493  // return value of write_pdu
494  return ret != -1 ? 0 : -1;
495 }

506 int
507 handle_remote_input(dchat_conf_t* cnf, int n)
508 {
509  dchat_pdu_t* pdu; // pdu read from contact file descriptor
510  char* txt_msg; // message used to store remote input
511  int ret; // return value
512  int len; // amount of bytes read
513  int fd; // file descriptor of a contact
514  fd = cnf->cl.contact[n].fd;
515 
516  // read pdu from file descriptor (-1 indicates error)
517  if ((len = read_pdu(fd, &pdu)) == -1)
518  {
519  log_msg(LOG_ERR, "Illegal PDU from contact %d", n);
520  return -1;
521  }
522  // EOF
523  else if (!len)
524  {
525  log_msg(LOG_INFO, "contact '%d' disconnected", n);
526  return 0;
527  }
528 
529  // the first pdu of a newly connected client has to be a
530  // "control/discover" otherwise raise an error and delete
531  // this contact
532  if (!cnf->cl.contact[n].lport && pdu->content_type != CT_CTRL_DISC)
533  {
534  log_msg(LOG_ERR, "Contact %d has not identfied himself", n);
535  return -1;
536  }
537  else if (!cnf->cl.contact[n].lport && pdu->content_type == CT_CTRL_DISC)
538  {
539  // set listening port of contact
540  cnf->cl.contact[n].lport = pdu->listen_port;
541  }
542 
543  /*
544  * == TEXT/PLAIN ==
545  */
546  if (pdu->content_type == CT_TXT_PLAIN)
547  {
548  // allocate memory for text message
549  if ((txt_msg = malloc(pdu->content_length + 1)) == NULL)
550  {
551  log_errno(LOG_ERR,
552  "handle_remote_input() failed - Could not allocate memory");
553  free_pdu(pdu);
554  return -1;
555  }
556 
557  // store bytes from pdu in txt_msg and terminate it
558  memcpy(txt_msg, pdu->content, pdu->content_length);
559  txt_msg[pdu->content_length] = '\0';
560  // print text message
561  print_dchat_msg(txt_msg, cnf->out_fd);
562  free(txt_msg);
563  }
564  /*
565  * == CONTROL/DISCOVER ==
566  */
567  else if (pdu->content_type == CT_CTRL_DISC)
568  {
569  // since dchat brings with the problem of duplicate contacts
570  // check if there are duplicate contacts in the contactlist
571  if ((ret = check_duplicates(cnf, n)) == -2)
572  {
573  // ERROR
574  log_msg(LOG_ERR, "check_duplicate_contact() failed");
575  }
576  // no duplicates
577  else if (ret == -1)
578  {
579  // log_msg(LOG_INFO, "no duplicates founds");
580  }
581  // duplicates found
582  else
583  {
584  log_msg(LOG_INFO, "Duplicate detected - Removing it (%d)", ret);
585  del_contact(cnf, ret); // delete duplicate
586  }
587 
588  // iterate through the content of the pdu containing
589  // the new contacts
590  if ((ret = receive_contacts(cnf, pdu)) == -1)
591  {
592  log_msg(LOG_WARN,
593  "handle_remote_input() - 'control/discover' failed: Not every new contact could be added");
594  }
595  else if (ret == 0)
596  {
597  // log_msg(LOG_INFO, "client %d does not know any other
598  // contacts we do not know about", n);
599  }
600  else
601  {
602  // log_msg(LOG_INFO, "%d new client(s) have been added",
603  // ret);
604  }
605  }
606  /*
607  * == UNKNOWN CONTENT-TYPE ==
608  */
609  else
610  {
611  log_msg(LOG_WARN,
612  "handle_remote_input(): Content-Type not implemented yet");
613  }
614 
615  free_pdu(pdu);
616  return len;
617 }

632 {

634  int n; // index of the contact in our contactlist
635

638  {

640  }
641  else
642  {
643  // add contact
644  if ((n = add_contact(cnf, s, da)) == -1)
645  {
646  log_errno(LOG_ERR,
647  "handle_local_conn_request() failed - Could not add contact");
648  return -1;
649  }
650  else
651  {
652  // set listening port of contact (we now it, since we
653  // connected to it)
654  if (ip_version(da) == 4)
655  {

657  ntohs(((struct sockaddr_in*) da)->sin_port);
658  }

660  {
661  cnf->cl.contact[n].lport =
662  ntohs(((struct sockaddr_in6*) da)->sin6_port);
663  }
664 
665  // send all our known contacts to the newly connected client
666  send_contacts(cnf, n);
667  }
668  }
669

722 void*
723 th_new_conn(dchat_conf_t* cnf)
724 {
725  struct sockaddr_storage da; // destination address to connect to
726  char c = '1'; // will be written to `cl_change`
727  int ret;
728 
729  for (;;)
730  {
731  // read from pipe to get new addresses
732  if ((ret = read(cnf->connect_fd[0], &da, sizeof(da))) == -1)
733  {
734  log_msg(LOG_WARN, "th_new_conn(): could not read from pipe");
735  }
736 
737  // EOF
738  if (!ret)
739  {
740  close(cnf->connect_fd[0]);
741  close(cnf->cl_change[1]);
742  break;
743  }
744 
745  // lock contactlist
746  pthread_mutex_lock(&cnf->cl.cl_mx);
747 
748  if (handle_local_conn_request(cnf, &da) == -1)
749  {
750  log_errno(LOG_WARN,
751  "th_new_conn(): Could not execute connection request successfully");
752  }
753  else if ((write(cnf->cl_change[1], &c, sizeof(c))) == -1)
754  {
755  log_msg(LOG_WARN, "th_new_conn(): could not write to pipe");
756  }
757 
758  // unlock contactlist
759  pthread_mutex_unlock(&cnf->cl.cl_mx);
760  }
761 
762  pthread_exit(NULL);
763 }

774 void*
775 th_new_input(dchat_conf_t* cnf)
776 {
777  char prompt[MAX_NICKNAME + 8]; // readline prompt contains nickname
778  char* line; // line read from stdin
779  int len; // length of line
780  // Assemble prompt for GNU readline
781  prompt[0] = '\0';
782  strncat(prompt, "Me (", 4);
783  strncat(prompt, cnf->me.name, MAX_NICKNAME);
784  strncat(prompt, "): ", 3);
785 
786  while (1)
787  {
788  // wait until user entered a line that can be read from stdin
789  line = readline(prompt);
790 
791  // EOF or user has entered "/exit"
792  if (line == NULL || !strcmp(line, "/exit"))
793  {
794  // since EOF is not a signal - raise SIGTERM
795  // to terminate this process
796  raise(SIGTERM);
797  break;
798  }
799  else
800  {
801  len = strlen(line);
802 
803  // user did not write anything -> just hit enter
804  if (len == 0)
805  {
806  len = 1;
807  write(cnf->user_input[1], &len, sizeof(int));
808  write(cnf->user_input[1], "\n", len);
809  }
810  else
811  {
812  // first write length of string
813  write(cnf->user_input[1], &len, sizeof(int));
814  // then write the bytes of the string
815  write(cnf->user_input[1], line, len);
816  }
817 
818  free(line);
819  }
820  }
821 
822  pthread_exit(NULL);
823 }

838 th_main_loop(dchat_conf_t* cnf)
839 {
840  fd_set rset; // list of readable file descriptors

892  }
893  }
894

896  // pipe
897  if (FD_ISSET(cnf->user_input[0], &rset))
898  {

901  read(cnf->user_input[0], &ret, sizeof(int));
902

905  {

908  }
909 
910  // allocate memory for the string entered from user
911  line = malloc(ret + 1);
912  // read string
913  read(cnf->user_input[0], line, ret);
914  line[ret] = '\0';
915  pthread_mutex_lock(&cnf->cl.cl_mx);
916 
917  // handle user input
918  if ((ret = handle_local_input(cnf, line)) == -1)
919  {
920  return -1;
921  }
922 
923  pthread_mutex_unlock(&cnf->cl.cl_mx);
924  free(line);
925  }

928  // accepted
929  if (FD_ISSET(cnf->acpt_fd, &rset))
930  {
931  nfds--;
932  pthread_mutex_lock(&cnf->cl.cl_mx);
933

936  {
937  return -1;
938  }
939  else
940  {
941  send_contacts(cnf, ret);
942  }
943 
944  pthread_mutex_unlock(&cnf->cl.cl_mx);
945  }
946 
947  // CHECK NEW CONN: check if user new connection has been added
948  if (FD_ISSET(cnf->cl_change[0], &rset))
949  {
950  nfds--;
951 
952  // !< EOF
953  if (!read(cnf->cl_change[0], &c, sizeof(c)))
954  {
955  close(cnf->cl_change[0]);
956  return 0;
957  }
958  }
959 
960  // CHECK CONTACTS: check file descriptors of contacts
961  // check if nfds is 0 => no contacts have written something
962  pthread_mutex_lock(&cnf->cl.cl_mx);
963 
964  for (i = 0; nfds && i < cnf->cl.cl_size; i++)
965  {
966  if (FD_ISSET(cnf->cl.contact[i].fd, &rset))
967  {
968  nfds--;
969 
970  // handle input from remote user
971  // -1 = error, 0 = EOF
972  if ((ret = handle_remote_input(cnf, i)) == -1 || ret == 0)
973  {
974  close(cnf->cl.contact[i].fd);
975  del_contact(cnf, i);
976  }
977  }
978  }
979 
980  pthread_mutex_unlock(&cnf->cl.cl_mx);
981  }
