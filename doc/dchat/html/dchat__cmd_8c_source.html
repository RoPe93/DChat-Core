46 int
47 parse_cmd_connect(dchat_conf_t* cnf, char* cmd)
48 {
49  struct sockaddr_storage ss;
50  char* address;
51  char* endptr;
52  int r;
53  char* tmpport;
54  int port = 0;
55 
56  // if the string contains more spaces the pointer is after the loop at a non-space char
57  for (; isspace(*cmd); cmd++);
58 
59  // is after the command no value the function returns
60  if (*cmd == '\0')
61  {
62  return -1;
63  }
64 
65  // the first of the command have to be the address
66  address = strtok_r(cmd, " ", &endptr);
67 
68  // is there no port the functions returns -1 for error
69  if ((tmpport = strtok_r(NULL, " \t\r\n", &endptr)) == NULL)
70  {
71  return -1;
72  }
73 
74  port = atoi(tmpport);
75 
76  // check if the port is in a valid range
77  if (port <= 0 || port > 65535)
78  {
79  return -3;
80  }
81 
82  memset(&ss, 0, sizeof(ss));
83 
84  // check if the address is in the IPv4 format
85  if (strchr(cmd, '.') != NULL)
86  {
87  // write the declared parameter to the address structure
88  r = inet_pton(AF_INET, address, &((struct sockaddr_in*) &ss)->sin_addr);
89  ((struct sockaddr*) &ss)->sa_family = AF_INET;
90  ((struct sockaddr_in*) &ss)->sin_port = htons(port);
91  }
92  // check if address is in IPv6 format
93  else if (strchr(cmd, ':') != NULL)
94  {
95  // write the declared parameter to the address structure
96  r = inet_pton(AF_INET6, address, &((struct sockaddr_in6*) &ss)->sin6_addr);
97  ((struct sockaddr*) &ss)->sa_family = AF_INET6;
98  ((struct sockaddr_in6*) &ss)->sin6_port = htons(port);
99  }
100  else
101  {
102  // if address is not in IPv4 or IPv6 format an error is returned
103  return -1;
104  }
105 
106  if (r != 1)
107  {
108  // is the address invalid -2 is returned
109  return -2;
110  }
111 
112  // the new contact is written to the connector pipe
113  if ((write(cnf->connect_fd[1], &ss, sizeof(ss)) == -1))
114  {
115  return -1;
116  }
117 
118  return 0;
119 }

149 parse_cmd_list(dchat_conf_t* cnf)
150 {
151  int i, count = 0;
152  char address[INET6_ADDRSTRLEN + 1];
153  unsigned short port;
154  struct sockaddr_storage* client_addr;
155 
156  for (i = 0; i < cnf->cl.cl_size; i++)
157  {
158  // check if entry is a valid connection
159  if (cnf->cl.contact[i].fd)
160  {
161  client_addr = &cnf->cl.contact[i].stor;
162 
163  // check the version of the ip address
164  if (ip_version(client_addr) == 4)
165  {
166  // get the address from the sockaddr_storage structure
167  inet_ntop(AF_INET, &(((struct sockaddr_in*) client_addr)->sin_addr), address,
168  INET_ADDRSTRLEN);
169  // get the port from the structure
170  port = ntohs(((struct sockaddr_in*) client_addr)->sin_port);
171  }
172  else if (ip_version(client_addr) == 6)
173  {
174  // get the address from the sockaddr_storage structure
175  inet_ntop(AF_INET, &(((struct sockaddr_in6*) client_addr)->sin6_addr), address,
176  INET_ADDRSTRLEN);
177  // get the port from the structure
178  port = ntohs(((struct sockaddr_in6*) client_addr)->sin6_port);
179  }
180  else
181  {
182  return -1;
183  }
184 
185  // print all available information about the connection
186  dprintf(cnf->out_fd, "\n\n"
187  " Contact................%d\n"
188  " Address-Tupel..........%s:%u\n"
189  " Listening-Port.........%u\n"
190  " Chatroom...............%d", i, address, port, cnf->cl.contact[i].lport,
191  cnf->cl.contact[i].chatrooms);
192  count++;
193  }
194  }
195 
196  // are there no contacts in the list a message will be printed
197  if (!count)
198  {
199  log_msg(LOG_INFO, "No contacts found in the contactlist");
200  }
201 
202  return 0;
203 }

213 parse_cmd(dchat_conf_t* cnf, char* buf)
214 {
215  int ret;
216 
217  //check if the string contains the command /connect
218  if (strncmp(buf, "/connect ", 9) == 0)
219  {
220  // to get the string without the command
221  buf += 9;
222 
223  if ((ret = parse_cmd_connect(cnf, buf)) == -1)
224  {
225  log_msg(LOG_WARN, "Wrong Syntax! Syntax of connect is: /connect <IP> <PORT>");
226  return -2;
227  }
228  else if (ret == -2)
229  {
230  log_msg(LOG_WARN, "Wrong IP address format");
231  return -2;
232  }
233  else if (ret == -3)
234  {
235  log_msg(LOG_WARN, "Invalid Portnumber");
236  return -2;
237  }
238  }
239  // check if the command contains exit and return 1
240  else if (strcmp(buf, "/exit") == 0)
241  {
242  raise(SIGTERM);
243  }
244  // check for the command help and call the appropriate function
245  else if (strcmp(buf, "/help") == 0)
246  {
247  parse_cmd_help();
248  }
249  // check for the command list and call the appropriate function
250  else if (strcmp(buf, "/list") == 0)
251  {
252  parse_cmd_list(cnf);
253  }
254  else
255  {
256  return -1;
257  }
258 
259  return 0;
