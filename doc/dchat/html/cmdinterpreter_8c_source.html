46 parse_cmd(dchat_conf_t* cnf, char* line)
47 {
48  cmds_t cmds;
49  char* cmd;
50  char* arg;
51  char* delim = " ";
52  int ret = -1;
53  char* line_temp;
54 
55  if (init_cmds(&cmds) == -1)
56  {
57  return -1;
58  }
59 
60  line_temp = malloc(strlen(line) + 1);
61 
62  if (line_temp == NULL)
63  {
64  fatal("Memory allocation for command line failed!");
65  }
66 
67  line_temp[0] = '\0';
68  strcat(line_temp, line);
69 
70  if ((cmd = strtok_r(line_temp, delim, &arg)) == NULL)
71  {
72  free(line_temp);
73  return -1;
74  }
75 
76  for (int i = 0; i < CMD_AMOUNT; i++)
77  {
78  if (!strcmp(cmd, cmds.cmd[i].cmd_name))
79  {
80  if ((ret = cmds.cmd[i].execute(cnf, arg)) == 1)
81  {
82  log_msg(LOG_NOTICE, "Command syntax: %s", cmds.cmd[i].syntax);
83  }
84 
85  break;
86  }
87  }
88 
89  free(line_temp);
90  return ret;
91 }

113

116  return -1;

118

120  memcpy(cmds->cmd, &temp, sizeof(temp));

155 con_exec(dchat_conf_t* cnf, char* arg)
156 {
157  char* address;
158  char* port_str;
159  int port;
160  char* endptr;
161  char* prefix;
162 
163  // if the string contains more spaces the pointer is after the loop at a non-space char
164  if (remove_leading_spaces(arg) == NULL)
165  {
166  return 1;
167  }
168 
169  address = strtok_r(arg, " ", &endptr);
170 
171  if ((port_str = strtok_r(NULL, " \t\r\n", &endptr)) == NULL)
172  {
173  return 1;
174  }
175 
176  port = (int) strtol(port_str, &endptr, 10);
177 
178  if (!is_valid_port(port) || *endptr != '\0')
179  {
180  log_msg(LOG_WARN, "Invalid port '%s'!", port_str);
181  return 1;
182  }
183 
184  if (!is_valid_onion(address))
185  {
186  log_msg(LOG_WARN, "Invalid onion-id '%s'!", address);
187  return 1;
188  }
189 
190  // write onion address to connector pipe
191  if (write(cnf->connect_fd[1], address, ONION_ADDRLEN) == -1)
192  {
193  return -1;
194  }
195 
196  if (write(cnf->connect_fd[1], &port, sizeof(uint16_t)) == -1)
197  {
198  return -1;
199  }
200 
201  return 0;
202 }
