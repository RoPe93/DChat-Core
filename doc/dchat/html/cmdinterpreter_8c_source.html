42 int
43 parse_cmd_connect(dchat_conf_t* cnf, char* cmd)
44 {
45  char* address;
46  char* port_str;
47  int port;
48  char* endptr;
49  char* prefix;
50 
51  // if the string contains more spaces the pointer is after the loop at a non-space char
52  for (; isspace(*cmd); cmd++);
53 
54  // empty string?
55  if (*cmd == '\0')
56  {
57  return 1;
58  }
59 
60  address = strtok_r(cmd, " ", &endptr);
61 
62  if ((port_str = strtok_r(NULL, " \t\r\n", &endptr)) == NULL)
63  {
64  return 1;
65  }
66 
67  port = (int) strtol(port_str, &endptr, 10);
68 
69  if (!is_valid_port(port) || *endptr != '\0')
70  {
71  log_msg(LOG_WARN, "Invalid port '%s'!", port_str);
72  return 1;
73  }
74 
75  if (!is_valid_onion(address))
76  {
77  log_msg(LOG_WARN, "Invalid onion-id '%s'!", address);
78  return 1;
79  }
80 
81  // write onion address to connector pipe
82  if (write(cnf->connect_fd[1], address, ONION_ADDRLEN) == -1)
83  {
84  return -1;
85  }
86 
87  if (write(cnf->connect_fd[1], &port, sizeof(uint16_t)) == -1)
88  {
89  return -1;
90  }
91 
92  return 0;
93 }

122 parse_cmd_list(dchat_conf_t* cnf)
123 {
124  int i;
125 
126  // are there no contacts in the list a message will be printed
127  if (!cnf->cl.used_contacts)
128  {
129  log_msg(LOG_INFO, "No contacts found in the contactlist");
130  }
131  else
132  {
133  for (i = 0; i < cnf->cl.cl_size; i++)
134  {
135  // check if entry is a valid connection
136  if (cnf->cl.contact[i].fd)
137  {
138  // print all available information about the connection

140  " Contact................%s\n"

142  " Listening-Port.........%u\n",
143  cnf->cl.contact[i].name,

159 int
160 parse_cmd(dchat_conf_t* cnf, char* buf)
161 {
162  int ret;
163 
164  //check if the string contains the command /connect
165  if (strncmp(buf, "/connect ", 9) == 0)
166  {
167  // to get the string without the command
168  buf += 9;
169 
170  if ((ret = parse_cmd_connect(cnf, buf)) == -1)
171  {
172  return -1;
173  }
174  else if (ret == 1)
175  {
176  log_msg(LOG_ERR, "Syntax: /connect <ONION-ID> <PORT>");
177  return 1;
178  }
179  }
180  // check for the command help and call the appropriate function
181  else if (strcmp(buf, "/help") == 0)
182  {
183  parse_cmd_help();
184  }
185  // check for the command list and call the appropriate function
186  else if (strcmp(buf, "/list") == 0)
187  {
188  parse_cmd_list(cnf);
189  }
190  else
191  {
192  return -1;
193  }
194 
195  return 0;
196 }
